33  template<class T>
34  Set<T> Dglap<T>::MatchObject(bool const& Up, int const& nf, Set<T> const& f) const
35  {
36  // Get matching conditions.
37  Set<Operator> MC = _MatchingConditions(Up, nf);
38 
39  // Create the object 'g' with the same convolution map of the
40  // matching conditions but containing the same objects of the
41  // input set of functions 'f'.
42  Set<T> g{MC.GetMap(),f.GetObjects()};
43 
44  // Convolute 'MC' and 'g'.
45  Set<T> MO = MC * g;
46  MO.SetMap(_SplittingFunctions((Up ? nf+1 : nf-1), 0).GetMap());
47 
48  // Return the convoluted object with the map on the next evolution
49  // step.
50  return MO;
51  }
