21  template<class T>
22  MatchedEvolution<T>::MatchedEvolution(T const& ObjRef,
23  double const& MuRef,
24  vector<double> const& Thresholds,
25  int const& nsteps):
26  _ObjRef(ObjRef),
27  _MuRef(MuRef),
28  _Thresholds(Thresholds),
29  _nsteps(nsteps)
30  {
31  // Compute squared reference scale.
32  _MuRef2 = pow(MuRef,2);
33 
34  // Compute log of the squared final scale.
35  _LogMuRef2 = log(_MuRef2);

37  // Compute squared thresholds.

51  template<class T>
52  T MatchedEvolution<T>::EvolveObject(int const& nf, double const& t0, double const& t1, T const& Obj0) const
53  {
54  // Return immediately "Obj0" if "t0" and "t1" are equal.
55  if (t0 == t1)
56  return Obj0;
57 
58  // Numerical solution of the evolution equation with fourth-order
59  // Runge-Kutta.
60  const auto dObj = rk4<T>([&](double const& t, T const& Obj)->T{ return Derivative(nf, t, Obj); });
61 
62  // Use "_nsteps" steps for the evolution.
63  double t = t0;
64  T Obj = Obj0;
65  const double dt = ( t1 - t0 ) / _nsteps;
66  for (int k = 0; k < _nsteps; k++)
67  {
68  Obj += dObj(t, Obj, dt);
69  t += dt;
70  }
71  return Obj;
72  }

75  template<class T>
76  T MatchedEvolution<T>::Evaluate(double const& mu) const
77  {
78  const double mu2 = pow(mu,2);
79  const double lmu2 = log(mu2);
80 
81  // Find initial and final number of flavours.
82  const int nfi = NF(_MuRef2, _Thresholds2);
83  const int nff = NF(mu2, _Thresholds2);
84 
85  // Don't do the matching is initial and final number of flavours
86  // are equal.
87  if (nfi == nff)
88  return EvolveObject(nfi, _LogMuRef2, lmu2, _ObjRef);
89 
90  // Direction of the evolution
91  const bool sgn = signbit(nfi - nff);
92 
93  // Create a vector of objects containing the object right above
94  // each threshold to make sure that every time a threshold is
95  // crossed a new object with a different convolution map is
96  // created (effective only when a "Set" object is evolved).
97  T vobj = _ObjRef;
98  double ti = _LogMuRef2;
99  double tf = _LogThresholds2[(sgn ? nfi : nfi-1)];
100  for (int inf = nfi; (sgn ? inf < nff : inf > nff); inf += (sgn ? 1 : -1))
101  {
102  vobj = MatchObject(sgn, inf, EvolveObject(inf, ti, tf, vobj));
103  ti = tf + (sgn ? 1 : -1) * eps8; // Add "eps8" to make sure to be above the threshold
104  tf = (sgn ? _LogThresholds2[min(inf+1,nff-1)] : _LogThresholds2[max(inf-2,nff-1)]);
105  }
106  return EvolveObject(nff, ti, lmu2, vobj);
107  }
