181 // Functions that fills in the SIDIS hard cross sections on two
182 // different grids.
183 SidisPolObjects InitializeSIDISpol(Grid const& gx, Grid const& gz, std::vector<double> const& Thresholds, std::vector<int> exclude = {})
184 {
185 report("Initializing SIDIS longitudinally polarised hard cross sections... ");
186 Timer t;
187
188 // Compute initial and final number of active flavours according
189 // to the vector of thresholds (it assumes that the threshold
190 // vector entries are ordered).
191 int nfi = 0;
192 int nff = Thresholds.size();
193 for (auto const& v : Thresholds)
194 if (v <= 0)
195 nfi++;
196
197 // Define object of the structure containing the DglapObjects.
198 SidisPolObjects SidisObj;
199
200 // ====================================================
201 // Compute SIDIS partonic cross sections.
202 // Expressions taken from Appendix C of hep-ph/9711387.
203 // ====================================================
204 // LO contribution
205 const Operator odeltax{gx, delta{}};
206 const Operator odeltaz{gz, delta{}};
207
208 if (std::find(exclude.begin(), exclude.end(), 0) == exclude.end()) SidisObj.G10qq.AddTerm({1, odeltax, odeltaz});
209
210 // NLO contributions
211 // G1
212 const Operator oD0x{gx, Dn{0}};
213 const Operator oD1x{gx, Dn{1}};
214 const Operator oD0z{gz, Dn{0}};
215 const Operator oD1z{gz, Dn{1}};
216
217 const double LLqq = - 16 * CF;
218 const double LSqq = 4 * CF;
219 const double SLqq = 4 * CF;
220 const double SSqq = 4 * CF;
221 const double K1qq = 4 * CF;
222 const double K2qq = 4 * CF;
223
224 const Operator orlqqx{gx, rlqqpol{}};
225 const Operator orsqqx{gx, rsqqpol{}};
226 const Operator or11qqx{gx, r11qqpol{}};
227 const Operator or21qqx{gx, r21qqpol{}};
228 const Operator olrqqz{gz, lrqqpol{}};
229 const Operator osrqqz{gz, srqqpol{}};
230 const Operator or12qqz{gz, r12qqpol{}};
231 const Operator or22qqz{gz, r22qqpol{}};
232
233 if (std::find(exclude.begin(), exclude.end(), 1) == exclude.end()) SidisObj.G11qq.AddTerm({LLqq, odeltax, odeltaz});
234 if (std::find(exclude.begin(), exclude.end(), 2) == exclude.end()) SidisObj.G11qq.AddTerm({LSqq, odeltax, oD1z });
235 if (std::find(exclude.begin(), exclude.end(), 3) == exclude.end()) SidisObj.G11qq.AddTerm({1, odeltax, olrqqz });
236 if (std::find(exclude.begin(), exclude.end(), 4) == exclude.end()) SidisObj.G11qq.AddTerm({SLqq, oD1x, odeltaz});
237 if (std::find(exclude.begin(), exclude.end(), 5) == exclude.end()) SidisObj.G11qq.AddTerm({SSqq, oD0x, oD0z });
238 if (std::find(exclude.begin(), exclude.end(), 6) == exclude.end()) SidisObj.G11qq.AddTerm({1, oD0x, osrqqz });
239 if (std::find(exclude.begin(), exclude.end(), 7) == exclude.end()) SidisObj.G11qq.AddTerm({1, orlqqx, odeltaz});
240 if (std::find(exclude.begin(), exclude.end(), 8) == exclude.end()) SidisObj.G11qq.AddTerm({1, orsqqx, oD0z });
241 if (std::find(exclude.begin(), exclude.end(), 9) == exclude.end()) SidisObj.G11qq.AddTerm({K1qq, or11qqx, or12qqz});
242 if (std::find(exclude.begin(), exclude.end(), 10) == exclude.end()) SidisObj.G11qq.AddTerm({K2qq, or21qqx, or22qqz});
243
244 const double K1gq = 2 * CF;
245 const double K2gq = - 4 * CF;
246
247 const Operator olrgqz{gz, lrgqpol{}};
248 const Operator osrgqz{gz, srgqpol{}};
249 const Operator or11gqx{gx, r11gqpol{}};
250 const Operator or12gqz{gz, r12gqpol{}};
251 const Operator or21gqx{gx, r21gqpol{}};
252 const Operator or22gqz{gz, r22gqpol{}};
253
254 if (std::find(exclude.begin(), exclude.end(), 11) == exclude.end()) SidisObj.G11gq.AddTerm({1, odeltax, olrgqz });
255 if (std::find(exclude.begin(), exclude.end(), 12) == exclude.end()) SidisObj.G11gq.AddTerm({1, oD0x, osrgqz });
256 if (std::find(exclude.begin(), exclude.end(), 13) == exclude.end()) SidisObj.G11gq.AddTerm({K1gq, or11gqx, or12gqz});
257 if (std::find(exclude.begin(), exclude.end(), 14) == exclude.end()) SidisObj.G11gq.AddTerm({K2gq, or21gqx, or22gqz});
258
259 const double K1qg = 1;
260
261 const Operator orlqgx{gx, rlqgpol{}};
262 const Operator orsqgx{gx, rsqgpol{}};
263 const Operator or11qgx{gx, r11qgpol{}};
264 const Operator or12qgz{gz, r12qgpol{}};
265
266 if (std::find(exclude.begin(), exclude.end(), 15) == exclude.end()) SidisObj.G11qg.AddTerm({1, orlqgx, odeltaz});
267 if (std::find(exclude.begin(), exclude.end(), 16) == exclude.end()) SidisObj.G11qg.AddTerm({1, orsqgx, oD0z });
268 if (std::find(exclude.begin(), exclude.end(), 17) == exclude.end()) SidisObj.G11qg.AddTerm({K1qg, or11qgx, or12qgz});
269
270 // ====================================================
271 // Approximated NNLO corrections derived from threshold
272 // resummation. They only contribute to the qq channel of g1.
273 // Expressions taken from Appendix B of arXiv:2109.00847.
274 // ====================================================
275 // Additional singular terms
276 const Operator oD2x{gx, Dn{2}};
277 const Operator oD3x{gx, Dn{3}};
278 const Operator oD2z{gz, Dn{2}};
279 const Operator oD3z{gz, Dn{3}};
280
281 // Non-singular (next-to-leading power) terms
282 const Operator ol1x{gx, ln{1}};
283 const Operator ol2x{gx, ln{2}};
284 const Operator ol3x{gx, ln{3}};
285 const Operator ol1z{gz, ln{1}};
286 const Operator ol2z{gz, ln{2}};
287 const Operator ol3z{gz, ln{3}};
288
289 // Constant function
290 const Operator oonex{gx, one{}};
291 const Operator oonez{gz, one{}};
292
293 // Overall constant (16 = 4^2 due to the different normalisation
294 // of the expansion parameter)
295 const double ovc = 16. * CF;
296
297 // Loop over nf
298 for (int nf = nfi; nf <= nff; nf++)
299 {
300 DoubleObject<Operator> cnf{};
301
302 // CF and leading-power terms (Eq. (62))
303 const double c1 = ovc * CF / 2.;
304 if (std::find(exclude.begin(), exclude.end(), 18) == exclude.end()) cnf.AddTerm({c1, odeltax, oD3z});
305 if (std::find(exclude.begin(), exclude.end(), 19) == exclude.end()) cnf.AddTerm({c1, oD3x, odeltaz});
306
307 const double c2 = ovc * CF * 3. / 2.;
308 if (std::find(exclude.begin(), exclude.end(), 20) == exclude.end()) cnf.AddTerm({c2, oD0x, oD2z});
309 if (std::find(exclude.begin(), exclude.end(), 21) == exclude.end()) cnf.AddTerm({c2, oD2x, oD0z});
310 if (std::find(exclude.begin(), exclude.end(), 22) == exclude.end()) cnf.AddTerm({2 * c2, oD1x, oD1z});
311
312 const double c3 = - ovc * CF * ( 4. + Pi2 / 3. );
313 if (std::find(exclude.begin(), exclude.end(), 23) == exclude.end()) cnf.AddTerm({c3, oD0x, oD0z});
314 if (std::find(exclude.begin(), exclude.end(), 24) == exclude.end()) cnf.AddTerm({c3, odeltax, oD1z});
315 if (std::find(exclude.begin(), exclude.end(), 25) == exclude.end()) cnf.AddTerm({c3, oD1x, odeltaz});
316
317 const double c4 = ovc * CF * 2. * zeta3;
318 if (std::find(exclude.begin(), exclude.end(), 26) == exclude.end()) cnf.AddTerm({c4, odeltax, oD0z});
319 if (std::find(exclude.begin(), exclude.end(), 27) == exclude.end()) cnf.AddTerm({c4, oD0x, odeltaz});
320
321 const double c5 = ovc * CF * ( 511. / 64. - 15. * zeta3 / 4. + 29. * Pi2 / 48. - 7. * Pi2 * Pi2 / 360. );
322 if (std::find(exclude.begin(), exclude.end(), 28) == exclude.end()) cnf.AddTerm({c5, odeltax, odeltaz});
323
324 // CF and next-to-leading-power terms (Eq. (63))
325 const double c6 = - ovc * CF * 3. / 2.;
326 if (std::find(exclude.begin(), exclude.end(), 29) == exclude.end()) cnf.AddTerm({c6, oD2x, oonez});
327 if (std::find(exclude.begin(), exclude.end(), 30) == exclude.end()) cnf.AddTerm({c6, oonex, oD2z});
328 if (std::find(exclude.begin(), exclude.end(), 31) == exclude.end()) cnf.AddTerm({2 * c6, oD1x, ol1z});
329 if (std::find(exclude.begin(), exclude.end(), 32) == exclude.end()) cnf.AddTerm({2 * c6, ol1x, oD1z});
330 if (std::find(exclude.begin(), exclude.end(), 33) == exclude.end()) cnf.AddTerm({c6, oD0x, ol2z});
331 if (std::find(exclude.begin(), exclude.end(), 34) == exclude.end()) cnf.AddTerm({c6, ol2x, oD0z});
332
333 const double c7 = - ovc * CF / 2.;
334 if (std::find(exclude.begin(), exclude.end(), 35) == exclude.end()) cnf.AddTerm({c7, odeltax, ol3z});
335 if (std::find(exclude.begin(), exclude.end(), 36) == exclude.end()) cnf.AddTerm({c7, ol3x, odeltaz});
336
337 // CA terms (Eq. (64))
338 const double c8 = - ovc * CA * 11. / 24.;
339 if (std::find(exclude.begin(), exclude.end(), 37) == exclude.end()) cnf.AddTerm({c8, odeltax, oD2z});
340 if (std::find(exclude.begin(), exclude.end(), 38) == exclude.end()) cnf.AddTerm({c8, oD2x, odeltaz});
341 if (std::find(exclude.begin(), exclude.end(), 39) == exclude.end()) cnf.AddTerm({2 * c8, oD0x, oD1z});
342 if (std::find(exclude.begin(), exclude.end(), 40) == exclude.end()) cnf.AddTerm({2 * c8, oD1x, oD0z});
343
344 const double c9 = ovc * CA * ( 67. / 36. - Pi2 / 12. );
345 if (std::find(exclude.begin(), exclude.end(), 41) == exclude.end()) cnf.AddTerm({c9, oD0x, oD0z});
346 if (std::find(exclude.begin(), exclude.end(), 42) == exclude.end()) cnf.AddTerm({c9, odeltax, oD1z});
347 if (std::find(exclude.begin(), exclude.end(), 43) == exclude.end()) cnf.AddTerm({c9, oD1x, odeltaz});
348
349 const double c10 = ovc * CA * ( 7. * zeta3 / 4. + 11. * Pi2 / 72. - 101. / 54. );
350 if (std::find(exclude.begin(), exclude.end(), 44) == exclude.end()) cnf.AddTerm({c10, odeltax, oD0z});
351 if (std::find(exclude.begin(), exclude.end(), 45) == exclude.end()) cnf.AddTerm({c10, oD0x, odeltaz});
352
353 const double c11 = ovc * CA * ( 43. * zeta3 / 12. + 17. * Pi2 * Pi2 / 720. - 1535. / 192. - 269. * Pi2 / 432. );
354 if (std::find(exclude.begin(), exclude.end(), 46) == exclude.end()) cnf.AddTerm({c11, odeltax, odeltaz});
355
356 // nf terms (Eq. (65))
357 const double c12 = ovc * nf / 12.;
358 if (std::find(exclude.begin(), exclude.end(), 47) == exclude.end()) cnf.AddTerm({c12, odeltax, oD2z});
359 if (std::find(exclude.begin(), exclude.end(), 48) == exclude.end()) cnf.AddTerm({c12, oD2x, odeltaz});
360 if (std::find(exclude.begin(), exclude.end(), 49) == exclude.end()) cnf.AddTerm({2 * c12, oD0x, oD1z});
361 if (std::find(exclude.begin(), exclude.end(), 50) == exclude.end()) cnf.AddTerm({2 * c12, oD1x, oD0z});
362
363 const double c13 = - ovc * nf * 5. / 18.;
364 if (std::find(exclude.begin(), exclude.end(), 51) == exclude.end()) cnf.AddTerm({c13, oD0x, oD0z});
365 if (std::find(exclude.begin(), exclude.end(), 52) == exclude.end()) cnf.AddTerm({c13, odeltax, oD1z});
366 if (std::find(exclude.begin(), exclude.end(), 53) == exclude.end()) cnf.AddTerm({c13, oD1x, odeltaz});
367
368 const double c14 = ovc * nf * ( 7. / 27. - Pi2 / 36. );
369 if (std::find(exclude.begin(), exclude.end(), 54) == exclude.end()) cnf.AddTerm({c14, odeltax, oD0z});
370 if (std::find(exclude.begin(), exclude.end(), 55) == exclude.end()) cnf.AddTerm({c14, oD0x, odeltaz});
371
372 const double c15 = ovc * nf * ( zeta3 / 6. + 19. * Pi2 / 216. + 127. / 96. );
373 if (std::find(exclude.begin(), exclude.end(), 56) == exclude.end()) cnf.AddTerm({c15, odeltax, odeltaz});
374
375 SidisObj.G12qq.insert({nf, cnf});
376 }
377 t.stop();
378
379 return SidisObj;
380 }
