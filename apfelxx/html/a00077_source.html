18  Operator::Operator(Grid const& gr, Expression const& expr, accuracy const& eps):
19  Integrator{},
20  LagrangeInterpolator{gr},
21  _grid(gr),
22  _expr(&expr)
23  {
24  // Scaling factor
25  _eta = _expr->eta();
26 
27  // Number of grids.
28  const int ng = _grid.nGrids();
29 
30  // Loop over the subgrids.
31  _Operator.resize(ng);
32  for (_ig = 0; _ig < ng; _ig++)
33  {
34  // Define the global subgrid.
35  const SubGrid& sg = _grid.GetSubGrid(_ig);
36 
37  // Get vector with the grid nodes.
38  const vector<double>& xg = sg.GetGrid();
39 
40  // Number of grid points.
41  const int nx = sg.nx();
42 
43  // Interpolation degree.
44  const int id = sg.InterDegree();
45 
46  // Limit the loop over "beta" according to whether "sg" is
47  // external.
48  const int gbound = ( sg.IsExternal() ? nx : 0 );
49 
50  _Operator[_ig].resize(gbound+1, nx+1, 0);
51  for (int beta = 0; beta <= gbound; beta++)
52  {
53  _xbeta = xg[beta];
54  // Local function. Can be computed outside the 'alpha'
55  // loop.
56  const double L = _eta * _expr->Local(_xbeta / xg[beta+1] / _eta);
57  for (_alpha = beta; _alpha <= nx; _alpha++)
58  {
59  // Weight of the subtraction term (it is a delta if
60  // _eta = 1).
61  _ws = Interpolant(_alpha, log(_xbeta / _eta), sg);
62 
63  // Given that the interpolation functions have
64  // discontinuos derivative on the nodes and are
65  // wiggly, it turns out that it is convenient to split
66  // the integrals into (id+1) intervals on each of
67  // which the integrand is smooth. This way, even
68  // though more integrals have to be computed, the
69  // integration converges faster.
70 
71  // Number of grid intervals we need to integrate over.
72  const int nmax = fmin(id,_alpha-beta) + 1;
73  const int nmin = fmax(0,_alpha+1-nx);
74 
75  // Integral.
76  double I = 0;
77  for (int jint = nmin; jint < nmax; jint++)
78  {
79  // Define integration bounds of the first
80  // iteration.
81  const double c = _xbeta / xg[_alpha-jint+1];
82  const double d = _xbeta / xg[_alpha-jint];
83 
84  // Compute the integral.
85  I += integrate(c, d, eps);
86  }
87  // Add the local part.
88  _Operator[_ig](beta,_alpha) = I + L * _ws;
89  }
90  }
91  }
92  }

95  double Operator::integrand(double const& x) const
96  {
97  const double z = x / _eta;
98  if (z >= 1)
99  return 0;
100  const double wr = Interpolant(_alpha, log(_xbeta / x), _grid.GetSubGrid(_ig));
101  return _expr->Regular(z) * wr + _expr->Singular(z) * ( wr - _ws );
102  }

113  Distribution Operator::operator *= (Distribution const& d) const
114  {
115  // Fast method to check that we are using the same Grid.
116  if (&this->_grid != &d.GetGrid())
117  throw runtime_error(error("Operator::operator*=", "Operator and Distribution grids do not match"));
118 
119  const vector<vector<double>>& sg = d.GetDistributionSubGrid();
120  vector<vector<double>> s(sg);
121  vector<double> j;
122 
123  // Compute the the distribution on the subgrids.
124  int const ng = _grid.nGrids(); //sg.size();
125  for (int ig = 0; ig < ng; ig++)
126  {
127  int const nx = _grid.GetSubGrid(ig).nx();
128 
129  // If the grid is external the product between the operator
130  // and the distribution has to be done in a standard way.
131  if (this->_grid.GetSubGrid(ig).IsExternal())
132  {
133  for (int alpha = 0; alpha <= nx; alpha++)
134  {
135  s[ig][alpha] = 0;
136  for (int beta = alpha; beta <= nx; beta++)
137  s[ig][alpha] += _Operator[ig](alpha,beta) * sg[ig][beta];
138  }
139  }
140  // If the grid is internal the product between the operator
141  // and the distribution has to be done exploiting the symmetry
142  // of the operator.
143  else
144  {
145  for (int alpha = 0; alpha <= nx; alpha++)
146  {
147  s[ig][alpha] = 0;
148  for (int beta = alpha; beta <= nx; beta++)
149  s[ig][alpha] += _Operator[ig](0,beta-alpha) * sg[ig][beta];
150  }
151  }
152 
153  // Set to zero the values above one.
154  for (int alpha = nx + 1; alpha < this->_grid.GetSubGrid(ig).InterDegree() + nx + 1; alpha++)
155  s[ig][alpha] = 0;
156 
157  // Compute the the distribution on the joint grid.
158  double xtrans;
159  if (ig < ng-1)
160  xtrans = this->_grid.GetSubGrid(ig+1).xMin();
161  else
162  xtrans = 1 + 2 * eps12;
163 
164  for (int alpha = 0; alpha <= nx; alpha++)
165  {
166  const double x = this->_grid.GetSubGrid(ig).GetGrid()[alpha];
167  if (xtrans - x < eps12)
168  break;
169  j.push_back(s[ig][alpha]);
170  }
171  }
172 
173  // Set to zero the values above one.
174  for (int alpha = 0; alpha < this->_grid.GetJointGrid().InterDegree(); alpha++)
175  j.push_back(0);
176 
177  return Distribution{d, s, j};
178  }

181  Operator& Operator::operator *= (Operator const& o)
182  {
183  // fast method to check that we are using the same Grid.
184  if (&this->_grid != &o.GetGrid())
185  throw runtime_error(error("Operator::operator*=", "Operators grid does not match"));
186 
187  const auto v = _Operator;
188 
189  const int ng = _grid.nGrids(); //sg.size();
190  for (int ig = 0; ig < ng; ig++)
191  {
192  const int nx = this->_grid.GetSubGrid(ig).nx();
193 
194  // If the grid is external the product between the operators
195  // has to be done in a standard way.
196  if (this->_grid.GetSubGrid(ig).IsExternal())
197  {
198  for (int alpha = 0; alpha <= nx; alpha++)
199  for (int beta = alpha; beta <= nx; beta++)
200  {
201  _Operator[ig](alpha,beta) = 0;
202  for (int gamma = alpha; gamma <= beta; gamma++)
203  _Operator[ig](alpha,beta) += v[ig](alpha,gamma) * o._Operator[ig](gamma,beta);
204  }
205  }
206  // If the grid is internal the product between the operators
207  // has to be done exploiting the symmetry of the operators.
208  else
209  {
210  _Operator[ig].resize(nx+1, nx+1, 0);
211  for (int alpha = 0; alpha <= nx; alpha++)
212  for (int beta = alpha; beta <= nx; beta++)
213  {
214  _Operator[ig](alpha,beta) = 0;
215  for (int gamma = alpha; gamma <= beta; gamma++)
216  _Operator[ig](alpha,beta) += v[ig](0,gamma-alpha) * o._Operator[ig](0,beta-gamma);
217  }
218  }
219  }
220  return *this;
221  }
