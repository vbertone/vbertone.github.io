23  template<class T>
24  TabulateObject<T>::TabulateObject(MatchedEvolution<T> &Object,
25  int const& nQ,
26  double const& QMin,
27  double const& QMax,
28  int const& InterDegree,
29  double const& Lambda):
30  QGrid<T>(nQ, QMin, QMax, InterDegree, Object.GetThresholds(), Lambda)
31  {
32  report("Tabulating object... ");
33  Timer t;
34 
35  // Save initial conditions.
36  const int nsteps = Object.GetNumberOfSteps();
37  const T ObjRef = Object.GetObjectRef();
38  const double MuRef = Object.GetMuRef();
39 
40  // Set number of steps of the RK algorith to 1.
41  Object.SetNumberOfSteps(1);
42 
43  // Find the point on the QGrid right below MuRef.
44  const int tQ = lower_bound(this->_Qg.begin(), this->_Qg.end(), MuRef) - this->_Qg.begin() - 1;
45 
46  // Loop on "_Qg" below "MuRef".
47  for (int iQ = tQ; iQ >= 0; iQ--)
48  {
49  T o = Object.Evaluate(this->_Qg[iQ]);
50  this->_GridValues.push_back(o);
51  Object.SetObjectRef(o);
52  Object.SetMuRef(this->_Qg[iQ]);
53  }
54 
55  // Reverse order of the elements.
56  reverse(this->_GridValues.begin(),this->_GridValues.end());
57 
58  // Loop on "_Qg" above "MuRef".
59  Object.SetObjectRef(ObjRef);
60  Object.SetMuRef(MuRef);
61  for (int iQ = tQ + 1; iQ < (int) this->_Qg.size(); iQ++)
62  {
63  T o = Object.Evaluate(this->_Qg[iQ]);
64  this->_GridValues.push_back(o);
65  Object.SetObjectRef(o);
66  Object.SetMuRef(this->_Qg[iQ]);
67  }
68 
69  // Reset initial conditions.
70  Object.SetNumberOfSteps(nsteps);
71  Object.SetObjectRef(ObjRef);
72  Object.SetMuRef(MuRef);
73 
74  t.stop();
75  }
