25  map<int,TmdObjects> InitializeTmdObjects(Grid const& g,
26  vector<double> const& Thresholds,
27  double const& IntEps)
28  {
29  report("Initializing TMD objects for matching and evolution... ");
30  Timer t;
31 
32  // Compute initial and final number of active flavours according
33  // to the vector of thresholds (it assumes that the thresholds
34  // vector entries are ordered).
35  int nfi = 0;
36  int nff = Thresholds.size();
37  for (auto const& v : Thresholds)
38  if (v <= 0)
39  nfi++;
40 
41  // ===============================================================
42  // LO matching functions operators.
43  map<int,Operator> MatchLO;
44  const Operator Id {g, Identity{}, IntEps};
45  const Operator Zero{g, Null{}, IntEps};
46  MatchLO.insert({EvolutionBasisQCD::PNSP, Id});
47  MatchLO.insert({EvolutionBasisQCD::PNSM, Id});
48  MatchLO.insert({EvolutionBasisQCD::PNSV, Id});
49  MatchLO.insert({EvolutionBasisQCD::PQQ, Id});
50  MatchLO.insert({EvolutionBasisQCD::PQG, Zero});
51  MatchLO.insert({EvolutionBasisQCD::PGQ, Zero});
52  MatchLO.insert({EvolutionBasisQCD::PGG, Id});
53 
54  // ===============================================================
55  // NLO matching functions operators.
56  // PDFs
57  map<int,map<int,Operator>> MatchPDFsNLO;
58  const Operator O1ns{g, C1ns{}, IntEps};
59  const Operator O1qg{g, C1qg{}, IntEps};
60  const Operator O1gq{g, C1gq{}, IntEps};
61  const Operator O1gg{g, C1gg{}, IntEps};
62  for (int nf = nfi; nf <= nff; nf++)
63  {
64  const Operator O1qgnf = nf * O1qg;
65  map<int,Operator> OM;
66  OM.insert({EvolutionBasisQCD::PNSP, O1ns});
67  OM.insert({EvolutionBasisQCD::PNSM, O1ns});
68  OM.insert({EvolutionBasisQCD::PNSV, O1ns});
69  OM.insert({EvolutionBasisQCD::PQQ, O1ns});
70  OM.insert({EvolutionBasisQCD::PQG, O1qgnf});
71  OM.insert({EvolutionBasisQCD::PGQ, O1gq});
72  OM.insert({EvolutionBasisQCD::PGG, O1gg});
73  MatchPDFsNLO.insert({nf,OM});
74  }
75 
76  // FFs
77  map<int,map<int,Operator>> MatchFFsNLO;
78  const Operator O1nsff{g, C1nsff{}, IntEps};
79  const Operator O1qgff{g, C1qgff{}, IntEps};
80  const Operator O1gqff{g, C1gqff{}, IntEps};
81  const Operator O1ggff{g, C1ggff{}, IntEps};
82  for (int nf = nfi; nf <= nff; nf++)
83  {
84  const Operator O1qgffnf = nf * O1qgff;
85  map<int,Operator> OM;
86  OM.insert({EvolutionBasisQCD::PNSP, O1nsff});
87  OM.insert({EvolutionBasisQCD::PNSM, O1nsff});
88  OM.insert({EvolutionBasisQCD::PNSV, O1nsff});
89  OM.insert({EvolutionBasisQCD::PQQ, O1nsff});
90  OM.insert({EvolutionBasisQCD::PQG, O1qgffnf});
91  OM.insert({EvolutionBasisQCD::PGQ, O1gqff});
92  OM.insert({EvolutionBasisQCD::PGG, O1ggff});
93  MatchFFsNLO.insert({nf,OM});
94  }
95 
96  // ===============================================================
97  // NNLO matching functions operators.
98  // PDFs
99  map<int,map<int,Operator>> MatchPDFsNNLO;
100  const Operator O2Vqqb{g, C2Vqqb{}, IntEps};
101  const Operator O2ps{g, C2ps{}, IntEps};
102  const Operator O2qg{g, C2qg{}, IntEps};
103  for (int nf = nfi; nf <= nff; nf++)
104  {
105  const Operator O2Vqq{g, C2Vqq{nf}, IntEps};
106  const Operator O2qgnf = nf * O2qg;
107  const Operator O2gq{g, C2gq{nf}, IntEps};
108  const Operator O2gg{g, C2gg{nf}, IntEps};
109  const Operator O2nsp = O2Vqq + O2Vqqb;
110  const Operator O2nsm = O2Vqq - O2Vqqb;
111  const Operator O2qq = O2nsp + nf * O2ps;
112  map<int,Operator> OM;
113  OM.insert({EvolutionBasisQCD::PNSP, O2nsp});
114  OM.insert({EvolutionBasisQCD::PNSM, O2nsm});
115  OM.insert({EvolutionBasisQCD::PNSV, O2nsm});
116  OM.insert({EvolutionBasisQCD::PQQ, O2qq});
117  OM.insert({EvolutionBasisQCD::PQG, O2qgnf});
118  OM.insert({EvolutionBasisQCD::PGQ, O2gq});
119  OM.insert({EvolutionBasisQCD::PGG, O2gg});
120  MatchPDFsNNLO.insert({nf,OM});
121  }
122 
123  // FFs
124  map<int,map<int,Operator>> MatchFFsNNLO;
125  const Operator O2Vqqbff{g, C2Vqqbff{}, IntEps};
126  const Operator O2psff{g, C2psff{}, IntEps};
127  const Operator O2qgff{g, C2qgff{}, IntEps};
128  for (int nf = nfi; nf <= nff; nf++)
129  {
130  const Operator O2Vqqff{g, C2Vqqff{nf}, IntEps};
131  const Operator O2qgffnf = nf * O2qgff;
132  const Operator O2gqff{g, C2gqff{nf}, IntEps};
133  const Operator O2ggff{g, C2ggff{nf}, IntEps};
134  const Operator O2nspff = O2Vqqff + O2Vqqbff;
135  const Operator O2nsmff = O2Vqqff - O2Vqqbff;
136  const Operator O2qqff = O2nspff + nf * O2psff;
137  map<int,Operator> OM;
138  OM.insert({EvolutionBasisQCD::PNSP, O2nspff});
139  OM.insert({EvolutionBasisQCD::PNSM, O2nsmff});
140  OM.insert({EvolutionBasisQCD::PNSV, O2nsmff});
141  OM.insert({EvolutionBasisQCD::PQQ, O2qqff});
142  OM.insert({EvolutionBasisQCD::PQG, O2qgffnf});
143  OM.insert({EvolutionBasisQCD::PGQ, O2gqff});
144  OM.insert({EvolutionBasisQCD::PGG, O2ggff});
145  MatchFFsNNLO.insert({nf,OM});
146  }
147 
148  // Define map containing the TmdObjects for each nf.
149  map<int,TmdObjects> TmdObj;
150 
151  // Construct sets of operators for each perturbative order for the
152  // matching functions. Initialize also coefficients of: beta
153  // function, GammaCusp, gammaV, and Collins-Soper anomalous
154  // dimensions.
155  map<int,map<int,Set<Operator>>> MatchingFunctionsPDFs;
156  for (int nf = nfi; nf <= nff; nf++)
157  {
158  TmdObjects obj;
159 
160  // Threshold
161  obj.Threshold = Thresholds[nf-1];
162 
163  // Beta function
164  obj.Beta.insert({0, beta0(nf)});
165  obj.Beta.insert({1, beta1(nf)});
166  obj.Beta.insert({2, beta2(nf)});
167 
168  // GammaCusp
169  obj.GammaCuspq.insert({0, 4 * CF * GammaCusp0()});
170  obj.GammaCuspq.insert({1, 4 * CF * GammaCusp1(nf)});
171  obj.GammaCuspq.insert({2, 4 * CF * GammaCusp2(nf)});
172  obj.GammaCuspg.insert({0, 4 * CA * GammaCusp0()});
173  obj.GammaCuspg.insert({1, 4 * CA * GammaCusp1(nf)});
174  obj.GammaCuspg.insert({2, 4 * CA * GammaCusp2(nf)});
175 
176  // GammaV
177  obj.GammaVq.insert({0, gammaVq0()});
178  obj.GammaVq.insert({1, gammaVq1(nf)});
179  obj.GammaVq.insert({2, gammaVq2(nf)});
180  obj.GammaVg.insert({0, gammaVg0(nf)});
181  obj.GammaVg.insert({1, gammaVg1(nf)});
182  obj.GammaVg.insert({2, gammaVg2(nf)});
183 
184  // CSd
185  valarray<double> d1{CSd10(), CSd11()};
186  valarray<double> d2{CSd20(nf), CSd21(nf), CSd22(nf)};
187  valarray<double> d3{CSd30(nf), CSd31(nf), CSd32(nf), CSd33(nf)};
188  obj.CSdq.insert({0, CF * d1});
189  obj.CSdq.insert({1, CF * d2});
190  obj.CSdq.insert({2, CF * d3});
191  obj.CSdg.insert({0, CA * d1});
192  obj.CSdg.insert({1, CA * d2});
193  obj.CSdg.insert({2, CA * d3});
194 
195  // Log(zeta) coefficients (for implementing the zeta-
196  // prescription)
197  obj.Lzetaq.insert({0, {Lzetaq10(), Lzetaq11()}});
198  obj.Lzetaq.insert({1, {Lzetaq20(nf), 0, Lzetaq22(nf)}});
199  obj.Lzetag.insert({0, {Lzetag10(nf), Lzetag11()}});
200  obj.Lzetag.insert({1, {Lzetag20(nf), 0, Lzetag22(nf)}});
201 
202  // Matching functions.
203  const EvolutionBasisQCD evb{nf};
204  obj.MatchingFunctionsPDFs.insert({0, Set<Operator>{evb, MatchLO}});
205  obj.MatchingFunctionsPDFs.insert({1, Set<Operator>{evb, MatchPDFsNLO.at(nf)}});
206  obj.MatchingFunctionsPDFs.insert({2, Set<Operator>{evb, MatchPDFsNNLO.at(nf)}});
207 
208  obj.MatchingFunctionsFFs.insert({0, Set<Operator>{evb, MatchLO}});
209  obj.MatchingFunctionsFFs.insert({1, Set<Operator>{evb, MatchFFsNLO.at(nf)}});
210  obj.MatchingFunctionsFFs.insert({2, Set<Operator>{evb, MatchFFsNNLO.at(nf)}});
211 
212  TmdObj.insert({nf, obj});
213  }
214  t.stop();
215 
216  return TmdObj;
217  }

220  function<Set<Distribution>(double const&, double const&, double const&)> BuildTmdPDFs(map<int,TmdObjects> const& TmdObj,
221  map<int,DglapObjects> const& DglapObj,
222  function<Set<Distribution>(double const&)> const& CollPDFs,
223  function<double(double const&, double const&)> const& fNP,
224  function<double(double const&)> const& Mu0b,
225  function<double(double const&)> const& Mub,
226  int const& PerturbativeOrder,
227  function<double(double const&)> const& Alphas,
228  double const& IntEps)
229  {
230  // Computed TMDs at the initial scale by convoluting PDFs,
231  // matching functions, and non-perturbative function.
232  const auto MatchedTmdPDFs = MatchTmdPDFs(TmdObj, DglapObj, CollPDFs, Mub, PerturbativeOrder, Alphas);
233 
234  // Compute TMD evolution factors.
235  const auto EvolFactors = EvolutionFactors(TmdObj, Mu0b, Mub, PerturbativeOrder, Alphas, IntEps);
236 
237  // Construct the function that returns the product and includes
238  // the non perturbative function.
239  const auto EvolvedTMDs = [=] (double const& b, double const& muf, double const& zetaf) -> Set<Distribution>
240  {
241  return EvolFactors(b, muf, zetaf) * ( [&] (double const& x) -> double{ return fNP(x, b); } * MatchedTmdPDFs(b) );
242  };
243 
244  return EvolvedTMDs;
245  }

248  function<Set<Distribution>(double const&, double const&, double const&)> BuildTmdFFs(map<int,TmdObjects> const& TmdObj,
249  map<int,DglapObjects> const& DglapObj,
250  function<Set<Distribution>(double const&)> const& CollFFs,
251  function<double(double const&, double const&)> const& fNP,
252  function<double(double const&)> const& Mu0b,
253  function<double(double const&)> const& Mub,
254  int const& PerturbativeOrder,
255  function<double(double const&)> const& Alphas,
256  double const& IntEps)
257  {
258  // Computed TMDs at the initial scale by convoluting FFs,
259  // matching functions, and non-perturbative function.
260  const auto MatchedTmdFFs = MatchTmdFFs(TmdObj, DglapObj, CollFFs, Mub, PerturbativeOrder, Alphas);
261 
262  // Compute TMD evolution factors.
263  const auto EvolFactors = EvolutionFactors(TmdObj, Mu0b, Mub, PerturbativeOrder, Alphas, IntEps);
264 
265  // Construct the function that returns the product and includes
266  // the non perturbative function. Include a factor 1/z^2 typical
267  // of FFs.
268  const auto EvolvedTMDs = [=] (double const& b, double const& muf, double const& zetaf) -> Set<Distribution>
269  {
270  return EvolFactors(b, muf, zetaf) * ( [&] (double const& z) -> double{ return fNP(z, b); } * MatchedTmdFFs(b) );
271  };
272 
273  return EvolvedTMDs;
274  }

277  function<Set<Distribution>(double const&)> MatchTmdPDFs(map<int,TmdObjects> const& TmdObj,
278  map<int,DglapObjects> const& DglapObj,
279  function<Set<Distribution>(double const&)> const& CollPDFs,
280  function<double(double const&)> const& Mub,
281  int const& PerturbativeOrder,
282  function<double(double const&)> const& Alphas)
283  {
284  // Retrieve thresholds from "TmdObj".
285  vector<double> thrs;
286  for(auto const& obj : TmdObj)
287  {
288  const int nf = obj.first;
289  const double thr = obj.second.Threshold;
290  if ((int) thrs.size() < nf)
291  thrs.resize(nf);
292  thrs[nf-1] = thr;
293  }
294 
295  // Define the LX fuction as in eq. (4.9) of arXiv:1604.07869.
296  const double C0 = 2 * exp(- emc);
297  const auto LX = [C0] (double const& mu, double const& b) -> double{ return 2 * log( b * mu / C0 ); };
298 
299  // Matching functions as functions of the absolute value of the
300  // impact parameter b.
301  function<Set<Operator>(double const&)> MatchFunc;
302  if (PerturbativeOrder == 0)
303  MatchFunc = [=] (double const& b) -> Set<Operator>
304  {
305  return TmdObj.at(NF(Mub(b),thrs)).MatchingFunctionsPDFs.at(0);
306  };
307  else if (PerturbativeOrder == 1)
308  MatchFunc = [=] (double const& b) -> Set<Operator>
309  {
310  const double mu = Mub(b);
311  const int nf = NF(mu,thrs);
312  const auto mf = TmdObj.at(nf).MatchingFunctionsPDFs;
313  const auto sf = DglapObj.at(nf).SplittingFunctions;
314  const double Lmu = LX(mu,b);
315  const double coup = Alphas(mu) / FourPi;
316  return mf.at(0) + coup * ( - Lmu * sf.at(0) + mf.at(1) );
317  };
318  else if (PerturbativeOrder == 2)
319  {
320  // Precompute set of operators on the O(as^2) bit that are
321  // proportional to the different powers of Lmu (see eq. (2.37)
322  // of https://arxiv.org/pdf/1706.01473.pdf).
323  map<int,Set<Operator>> SetLcoef;
324  map<int,Set<Operator>> SetL2coef;
325  for (auto const& to : TmdObj)
326  {
327  const int nf = to.first;
328  const double b0 = to.second.Beta.at(0);
329  const auto mf = to.second.MatchingFunctionsPDFs;
330  const auto sf = DglapObj.at(nf).SplittingFunctions;
331 
332  // Construct matricial product of P0 * P0 and C1 * P0 (see
333  // eq. (B.15) https://arxiv.org/pdf/1706.01473.pdf).
334  const auto P0 = sf.at(0);
335  const auto P1 = sf.at(1);
336  const auto C1 = mf.at(1);
337  const auto ConvMap = P0.GetMap();
338 
339  map<int,Operator> MapL2coef;
340  MapL2coef.insert({EvolutionBasisQCD::PNSP, ( P0.at(0) * P0.at(0) - b0 * P0.at(0) ) / 2});
341  MapL2coef.insert({EvolutionBasisQCD::PNSM, ( P0.at(1) * P0.at(1) - b0 * P0.at(1) ) / 2});
342  MapL2coef.insert({EvolutionBasisQCD::PNSV, ( P0.at(2) * P0.at(2) - b0 * P0.at(2) ) / 2});
343  MapL2coef.insert({EvolutionBasisQCD::PQQ, ( P0.at(3) * P0.at(3) + P0.at(4) * P0.at(5) - b0 * P0.at(3) ) / 2});
344  MapL2coef.insert({EvolutionBasisQCD::PQG, ( P0.at(3) * P0.at(4) + P0.at(4) * P0.at(6) - b0 * P0.at(4) ) / 2});
345  MapL2coef.insert({EvolutionBasisQCD::PGQ, ( P0.at(5) * P0.at(3) + P0.at(6) * P0.at(5) - b0 * P0.at(5) ) / 2});
346  MapL2coef.insert({EvolutionBasisQCD::PGG, ( P0.at(5) * P0.at(4) + P0.at(6) * P0.at(6) - b0 * P0.at(6) ) / 2});
347 
348  map<int,Operator> MapLcoef;
349  MapLcoef.insert({EvolutionBasisQCD::PNSP, P1.at(0) + C1.at(0) * P0.at(0) - b0 * C1.at(0)});
350  MapLcoef.insert({EvolutionBasisQCD::PNSM, P1.at(1) + C1.at(1) * P0.at(1) - b0 * C1.at(1)});
351  MapLcoef.insert({EvolutionBasisQCD::PNSV, P1.at(2) + C1.at(2) * P0.at(2) - b0 * C1.at(2)});
352  MapLcoef.insert({EvolutionBasisQCD::PQQ, P1.at(3) + C1.at(3) * P0.at(3) + C1.at(4) * P0.at(5) - b0 * C1.at(3)});
353  MapLcoef.insert({EvolutionBasisQCD::PQG, P1.at(4) + C1.at(3) * P0.at(4) + C1.at(4) * P0.at(6) - b0 * C1.at(4)});
354  MapLcoef.insert({EvolutionBasisQCD::PGQ, P1.at(5) + C1.at(5) * P0.at(3) + C1.at(6) * P0.at(5) - b0 * C1.at(5)});
355  MapLcoef.insert({EvolutionBasisQCD::PGG, P1.at(6) + C1.at(5) * P0.at(4) + C1.at(6) * P0.at(6) - b0 * C1.at(6)});
356 
357  SetL2coef.insert({nf,Set<Operator>{ConvMap, MapL2coef}});
358  SetLcoef.insert({nf,Set<Operator>{ConvMap, MapLcoef}});
359  }
360 
361  // Now contruct the actual matching-function function.
362  MatchFunc = [=] (double const& b) -> Set<Operator>
363  {
364  const double mu = Mub(b);
365  const int nf = NF(mu,thrs);
366  const auto mf = TmdObj.at(nf).MatchingFunctionsPDFs;
367  const auto sf = DglapObj.at(nf).SplittingFunctions;
368  const double Lmu = LX(mu,b);
369  const double coup = Alphas(mu) / FourPi;
370  const auto nlo = mf.at(1) - Lmu * sf.at(0);
371  const auto nnlo = mf.at(2) - Lmu * ( SetLcoef.at(nf) - Lmu * SetL2coef.at(nf) );
372  return mf.at(0) + coup * ( nlo + coup * nnlo );
373  };
374  }
375 
376  // Construct function that returns the product of matching
377  // functions and collinear FFs.
378  const auto MatchedTMDs = [=] (double const& b) -> Set<Distribution>
379  {
380  return MatchFunc(b) * CollPDFs(Mub(b));
381  };
382 
383  return MatchedTMDs;
384  }

387  function<Set<Distribution>(double const&)> MatchTmdFFs(map<int,TmdObjects> const& TmdObj,
388  map<int,DglapObjects> const& DglapObj,
389  function<Set<Distribution>(double const&)> const& CollFFs,
390  function<double(double const&)> const& Mub,
391  int const& PerturbativeOrder,
392  function<double(double const&)> const& Alphas)
393  {
394  // Retrieve thresholds from "TmdObj".
395  vector<double> thrs;
396  for(auto const& obj : TmdObj)
397  {
398  const int nf = obj.first;
399  const double thr = obj.second.Threshold;
400  if ((int) thrs.size() < nf)
401  thrs.resize(nf);
402  thrs[nf-1] = thr;
403  }
404 
405  // Define the LX fuction as in eq. (4.9) of arXiv:1604.07869.
406  const double C0 = 2 * exp(- emc);
407  const auto LX = [C0] (double const& mu, double const& b) -> double{ return 2 * log( b * mu / C0 ); };
408 
409  // Matching functions as functions of the absolute value of the
410  // impact parameter b.
411  function<Set<Operator>(double const&)> MatchFunc;
412  if (PerturbativeOrder == 0)
413  MatchFunc = [=] (double const& b) -> Set<Operator>
414  {
415  return TmdObj.at(NF(Mub(b),thrs)).MatchingFunctionsFFs.at(0);
416  };
417  else if (PerturbativeOrder == 1)
418  MatchFunc = [=] (double const& b) -> Set<Operator>
419  {
420  const double mu = Mub(b);
421  const int nf = NF(mu,thrs);
422  const auto mf = TmdObj.at(nf).MatchingFunctionsFFs;
423  const auto sf = DglapObj.at(nf).SplittingFunctions;
424  const double Lmu = LX(mu,b);
425  const double coup = Alphas(mu) / FourPi;
426  return mf.at(0) + coup * ( - Lmu * sf.at(0) + mf.at(1) );
427  };
428  else if (PerturbativeOrder == 2)
429  {
430  // Precompute set of operators on the O(as^2) bit that are
431  // proportional to the different powers of Lmu (see eq. (2.37)
432  // of https://arxiv.org/pdf/1706.01473.pdf).
433  map<int,Set<Operator>> SetLcoef;
434  map<int,Set<Operator>> SetL2coef;
435  for (auto const& to : TmdObj)
436  {
437  const int nf = to.first;
438  const double b0 = to.second.Beta.at(0);
439  const auto mf = to.second.MatchingFunctionsFFs;
440  const auto sf = DglapObj.at(nf).SplittingFunctions;
441 
442  // Construct matricial product of P0 * P0 and C1 * P0 (see
443  // eq. (B.15) https://arxiv.org/pdf/1706.01473.pdf).
444  const auto P0 = sf.at(0);
445  const auto P1 = sf.at(1);
446  const auto C1 = mf.at(1);
447  const auto ConvMap = P0.GetMap();
448 
449  map<int,Operator> MapL2coef;
450  MapL2coef.insert({EvolutionBasisQCD::PNSP, ( P0.at(0) * P0.at(0) - b0 * P0.at(0) ) / 2});
451  MapL2coef.insert({EvolutionBasisQCD::PNSM, ( P0.at(1) * P0.at(1) - b0 * P0.at(1) ) / 2});
452  MapL2coef.insert({EvolutionBasisQCD::PNSV, ( P0.at(2) * P0.at(2) - b0 * P0.at(2) ) / 2});
453  MapL2coef.insert({EvolutionBasisQCD::PQQ, ( P0.at(3) * P0.at(3) + P0.at(4) * P0.at(5) - b0 * P0.at(3) ) / 2});

471  // Now contruct the actual matching-function function.
472  MatchFunc = [=] (double const& b) -> Set<Operator>
473  {
474  const double mu = Mub(b);
475  const int nf = NF(mu,thrs);
476  const auto mf = TmdObj.at(nf).MatchingFunctionsFFs;
477  const auto sf = DglapObj.at(nf).SplittingFunctions;
478  const double Lmu = LX(mu,b);
479  const double coup = Alphas(mu) / FourPi;
480  const auto nlo = mf.at(1) - Lmu * sf.at(0);
481  const auto nnlo = mf.at(2) - Lmu * ( SetLcoef.at(nf) - Lmu * SetL2coef.at(nf) );
482  return mf.at(0) + coup * ( nlo + coup * nnlo );
483  };
484  }

487  // functions and collinear FFs.
488  const auto MatchedTMDs = [=] (double const& b) -> Set<Distribution>
489  {
490  return MatchFunc(b) * CollFFs(Mub(b));
491  };

497  function<vector<double>(double const&, double const&, double const&)> EvolutionFactors(map<int,TmdObjects> const& TmdObj,
498  function<double(double const&)> const& Mu0b,
499  function<double(double const&)> const& Mub,
500  int const& PerturbativeOrder,
501  function<double(double const&)> const& Alphas,
502  double const& IntEps)
503  {
504  // Retrieve thresholds from "TmdObj".
505  vector<double> thrs;
506  for(auto const& obj : TmdObj)
507  {
508  const int nf = obj.first;
509  const double thr = obj.second.Threshold;
510  if ((int) thrs.size() < nf)
511  thrs.resize(nf);
512  thrs[nf-1] = thr;
513  }
514 
515  // Define the LX fuction as in eq. (4.9) of arXiv:1604.07869.
516  const double C0 = 2 * exp(- emc);
517  const auto LX = [C0] (double const& mu, double const& b) -> double{ return 2 * log( b * mu / C0 ); };
518 
519  // Create functions needed for the TMD evolution.
520  function<double(double const&)> gammaVq;
521  function<double(double const&)> gammaVg;
522  function<double(double const&)> GammaCuspq;
523  function<double(double const&)> GammaCuspg;
524  function<double(double const&, double const&)> DCSq;
525  function<double(double const&, double const&)> DCSg;
526  function<double(double const&, double const&)> zetaq;
527  function<double(double const&, double const&)> zetag;
528  // LL
529  if (PerturbativeOrder == 0)
530  {
531  gammaVq = [=] (double const&) -> double{ return 0; };
532  gammaVg = [=] (double const&) -> double{ return 0; };
533  GammaCuspq = [=] (double const& mu) -> double
534  {
535  const double coup = Alphas(mu) / FourPi;
536  return coup * TmdObj.at(NF(mu,thrs)).GammaCuspq.at(0);
537  };
538  GammaCuspg = [=] (double const& mu) -> double
539  {
540  const double coup = Alphas(mu) / FourPi;
541  return coup * TmdObj.at(NF(mu,thrs)).GammaCuspg.at(0);
542  };
543  DCSq = [=] (double const& mu, double const& b) -> double
544  {
545  const auto d = TmdObj.at(NF(mu,thrs)).CSdq;
546  const double coup = Alphas(mu) / FourPi;
547  const double Lmu = LX(mu,b);
548  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
549  return coup * lo;
550  };
551  DCSg = [=] (double const& mu, double const& b) -> double
552  {
553  const auto d = TmdObj.at(NF(mu,thrs)).CSdg;
554  const double coup = Alphas(mu) / FourPi;
555  const double Lmu = LX(mu,b);
556  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
557  return coup * lo;
558  };
559  zetaq = [=] (double const& mu, double const&) -> double{ return mu * mu; };
560  zetag = [=] (double const& mu, double const&) -> double{ return mu * mu; };
561  }
562  // NLL
563  else if (PerturbativeOrder == 1)
564  {
565  gammaVq = [=] (double const& mu) -> double
566  {
567  const double coup = Alphas(mu) / FourPi;
568  return coup * TmdObj.at(NF(mu,thrs)).GammaVq.at(0);
569  };
570  gammaVg = [=] (double const& mu) -> double
571  {
572  const double coup = Alphas(mu) / FourPi;
573  return coup * TmdObj.at(NF(mu,thrs)).GammaVg.at(0);
574  };
575  GammaCuspq = [=] (double const& mu) -> double
576  {
577  const auto gc = TmdObj.at(NF(mu,thrs)).GammaCuspq;
578  const double coup = Alphas(mu) / FourPi;
579  return coup * ( gc.at(0) + coup * gc.at(1) );
580  };
581  GammaCuspg = [=] (double const& mu) -> double
582  {
583  const auto gc = TmdObj.at(NF(mu,thrs)).GammaCuspg;
584  const double coup = Alphas(mu) / FourPi;
585  return coup * ( gc.at(0) + coup * gc.at(1) );
586  };
587  DCSq = [=] (double const& mu, double const& b) -> double
588  {
589  const auto d = TmdObj.at(NF(mu,thrs)).CSdq;
590  const double coup = Alphas(mu) / FourPi;
591  const double Lmu = LX(mu,b);
592  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
593  const double nlo = d.at(1)[0] + Lmu * ( d.at(1)[1] + Lmu * d.at(1)[2] );
594  return coup * ( lo + coup * nlo );
595  };
596  DCSg = [=] (double const& mu, double const& b) -> double
597  {
598  const auto d = TmdObj.at(NF(mu,thrs)).CSdg;
599  const double coup = Alphas(mu) / FourPi;
600  const double Lmu = LX(mu,b);
601  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
602  const double nlo = d.at(1)[0] + Lmu * ( d.at(1)[1] + Lmu * d.at(1)[2] );
603  return coup * ( lo + coup * nlo );
604  };
605  zetaq = [=] (double const& mu, double const& b) -> double
606  {
607  const auto lz = TmdObj.at(NF(mu,thrs)).Lzetaq;
608  const double Lmu = LX(mu,b);
609  const double lo = lz.at(0)[0] + Lmu * lz.at(0)[1];
610  const double lzeta = lo;
611  return 4 * exp( - lzeta + Lmu - 2 * emc ) / b / b;
612  };
613  zetag = [=] (double const& mu, double const& b) -> double
614  {
615  const auto lz = TmdObj.at(NF(mu,thrs)).Lzetag;
616  const double Lmu = LX(mu,b);
617  const double lo = lz.at(0)[0] + Lmu * lz.at(0)[1];
618  const double lzeta = lo;
619  return 4 * exp( - lzeta + Lmu - 2 * emc ) / b / b;
620  };
621  }
622  // NNLL
623  else if (PerturbativeOrder == 2)
624  {
625  gammaVq = [=] (double const& mu) -> double
626  {
627  const auto gv = TmdObj.at(NF(mu,thrs)).GammaVq;
628  const double coup = Alphas(mu) / FourPi;
629  return coup * ( gv.at(0) + coup * gv.at(1) );
630  };
631  gammaVg = [=] (double const& mu) -> double
632  {
633  const auto gv = TmdObj.at(NF(mu,thrs)).GammaVg;
634  const double coup = Alphas(mu) / FourPi;
635  return coup * ( gv.at(0) + coup * gv.at(1) );
636  };
637  GammaCuspq = [=] (double const& mu) -> double
638  {
639  const auto gc = TmdObj.at(NF(mu,thrs)).GammaCuspq;
640  const double coup = Alphas(mu) / FourPi;
641  return coup * ( gc.at(0) + coup * ( gc.at(1) + coup * gc.at(2) ) );
642  };
643  GammaCuspg = [=] (double const& mu) -> double
644  {
645  const auto gc = TmdObj.at(NF(mu,thrs)).GammaCuspg;
646  const double coup = Alphas(mu) / FourPi;
647  return coup * ( gc.at(0) + coup * ( gc.at(1) + coup * gc.at(2) ) );
648  };
649  DCSq = [=] (double const& mu, double const& b) -> double
650  {
651  const auto d = TmdObj.at(NF(mu,thrs)).CSdq;
652  const double coup = Alphas(mu) / FourPi;
653  const double Lmu = LX(mu,b);
654  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
655  const double nlo = d.at(1)[0] + Lmu * ( d.at(1)[1] + Lmu * d.at(1)[2] );
656  const double nnlo = d.at(2)[0] + Lmu * ( d.at(2)[1] + Lmu * ( d.at(2)[2] + Lmu * d.at(2)[3] ) );
657  return coup * ( lo + coup * ( nlo + coup * nnlo ) );
658  };
659  DCSg = [=] (double const& mu, double const& b) -> double
660  {
661  const auto d = TmdObj.at(NF(mu,thrs)).CSdg;
662  const double coup = Alphas(mu) / FourPi;
663  const double Lmu = LX(mu,b);
664  const double lo = d.at(0)[0] + Lmu * d.at(0)[1];
665  const double nlo = d.at(1)[0] + Lmu * ( d.at(1)[1] + Lmu * d.at(1)[2] );
666  const double nnlo = d.at(2)[0] + Lmu * ( d.at(2)[1] + Lmu * ( d.at(2)[2] + Lmu * d.at(2)[3] ) );
667  return coup * ( lo + coup * ( nlo + coup * nnlo ) );
668  };
669  zetaq = [=] (double const& mu, double const& b) -> double
670  {
671  const auto lz = TmdObj.at(NF(mu,thrs)).Lzetaq;
672  const double coup = Alphas(mu) / FourPi;
673  const double Lmu = LX(mu,b);
674  const double lo = lz.at(0)[0] + Lmu * lz.at(0)[1];
675  const double nlo = lz.at(1)[0] + Lmu * Lmu * lz.at(1)[2];
676  const double lzeta = lo + coup * nlo;
677  return 4 * exp( - lzeta + Lmu - 2 * emc ) / b / b;
678  };
679  zetag = [=] (double const& mu, double const& b) -> double
680  {
681  const auto lz = TmdObj.at(NF(mu,thrs)).Lzetag;
682  const double coup = Alphas(mu) / FourPi;
683  const double Lmu = LX(mu,b);
684  const double lo = lz.at(0)[0] + Lmu * lz.at(0)[1];
685  const double nlo = lz.at(1)[0] + Lmu * Lmu * lz.at(1)[2];
686  const double lzeta = lo + coup * nlo;
687  return 4 * exp( - lzeta + Lmu - 2 * emc ) / b / b;
688  };
689  }
690 
691  // Define the integrands.
692  Integrator I1q{[=] (double const& mu) -> double{ return ( - gammaVq(mu) + 2 * log(mu) * GammaCuspq(mu) ) / mu; }};
693  Integrator I1g{[=] (double const& mu) -> double{ return ( - gammaVg(mu) + 2 * log(mu) * GammaCuspg(mu) ) / mu; }};
694  Integrator I2q{[=] (double const& mu) -> double{ return GammaCuspq(mu) / mu; }};
695  Integrator I2g{[=] (double const& mu) -> double{ return GammaCuspg(mu) / mu; }};
696 
697  // Construct function that returns the product of: matching
698  // functions, PDFs, NP function, and evolution factors, i.e. the
699  // set of evolved TMDs (times x).
700  const auto EvolFactors = [=] (double const& b, double const& muf, double const& zetaf) -> vector<double>
701  {
702  // Define relevant scales
703  const double mu0 = Mu0b(b);
704  const double mui = Mub(b);
705  const double zetaiq = zetaq(mui,b);
706  const double zetaig = zetag(mui,b);
707 
708  // Compute argument of the exponent of the evolution factors.
709  const double LRq = I1q.integrate(mui, muf, thrs, IntEps)
710  - I2q.integrate(mu0, muf, thrs, IntEps) * log(zetaf)
711  + I2q.integrate(mu0, mui, thrs, IntEps) * log(zetaiq);
712  const double LRg = I1g.integrate(mui, muf, thrs, IntEps)
713  - I2g.integrate(mu0, muf, thrs, IntEps) * log(zetaf)
714  + I2g.integrate(mu0, mui, thrs, IntEps) * log(zetaig);
715 
716  // Compute the evolution factors.
717  const double Rq = exp( LRq - DCSq(mu0,b) * log( zetaf / zetaiq ) );
718  const double Rg = exp( LRg - DCSg(mu0,b) * log( zetaf / zetaig ) );
719 
720  // Return vector of evolution factors.
721  return vector<double>{Rg, Rq, Rq, Rq, Rq, Rq, Rq, Rq, Rq, Rq, Rq, Rq, Rq};
722  };
723 
724  return EvolFactors;
725  }
