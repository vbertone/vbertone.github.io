23  map<int,double> PhysToQCDEv(map<int,double> const& InPhysMap)
24  {
25  // Call function in the physical basis.
26  map<int,double> PhysMap = InPhysMap;
27 
28  // Fill in keys that don't exist.
29  // Gluon (assumes that the ID is 21).
30  if (PhysMap.find(0) == PhysMap.end())
31  PhysMap[0] = PhysMap[21];
32 
33  // Quarks (Fill in with zero if they don't exist).
34  for (int i = -6; i <= 6; i++)
35  if (PhysMap.find(i) == PhysMap.end())
36  PhysMap[i] = 0;
37 
38  // Fill in map in the QCD evolution basis. It attumes that the
39  // gluon has key zero and all keys from -6 to 6 exist.
40  map<int,double> QCDEvMap;
41  QCDEvMap[0] = PhysMap.at(0);
42  QCDEvMap[1] =
43  + PhysMap.at(1) + PhysMap.at(-1)
44  + PhysMap.at(2) + PhysMap.at(-2)
45  + PhysMap.at(3) + PhysMap.at(-3)
46  + PhysMap.at(4) + PhysMap.at(-4)
47  + PhysMap.at(5) + PhysMap.at(-5)
48  + PhysMap.at(6) + PhysMap.at(-6);
49  QCDEvMap[2] =
50  + PhysMap.at(1) - PhysMap.at(-1)
51  + PhysMap.at(2) - PhysMap.at(-2)
52  + PhysMap.at(3) - PhysMap.at(-3)
53  + PhysMap.at(4) - PhysMap.at(-4)
54  + PhysMap.at(5) - PhysMap.at(-5)
55  + PhysMap.at(6) - PhysMap.at(-6);
56  QCDEvMap[3] =
57  + PhysMap.at(2) + PhysMap.at(-2)
58  - ( PhysMap.at(1) + PhysMap.at(-1) );
59  QCDEvMap[4] =
60  + PhysMap.at(2) - PhysMap.at(-2)
61  - ( PhysMap.at(1) - PhysMap.at(-1) );
62  QCDEvMap[5] =
63  + PhysMap.at(1) + PhysMap.at(-1)
64  + PhysMap.at(2) + PhysMap.at(-2)
65  - 2 * ( PhysMap.at(3) + PhysMap.at(-3) );
66  QCDEvMap[6] =
67  + PhysMap.at(1) - PhysMap.at(-1)
68  + PhysMap.at(2) - PhysMap.at(-2)
69  - 2 * ( PhysMap.at(3) - PhysMap.at(-3) );
70  QCDEvMap[7] =
71  + PhysMap.at(1) + PhysMap.at(-1)
72  + PhysMap.at(2) + PhysMap.at(-2)
73  + PhysMap.at(3) + PhysMap.at(-3)
74  - 3 * ( PhysMap.at(4) + PhysMap.at(-4) );
75  QCDEvMap[8] =
76  + PhysMap.at(1) - PhysMap.at(-1)
77  + PhysMap.at(2) - PhysMap.at(-2)
78  + PhysMap.at(3) - PhysMap.at(-3)
79  - 3 * ( PhysMap.at(4) - PhysMap.at(-4) );

82  + PhysMap.at(2) + PhysMap.at(-2)
83  + PhysMap.at(3) + PhysMap.at(-3)

86  QCDEvMap[10] =

111  map<int,double> QCDEvToPhys(map<int,double> const& QCDEvMap)
112  {
113  // Fill in map in the physical basis. It attumes that the gluon
114  // has key zero and all keys from 0 to 12 exist.
115  map<int,double> PhysMap;
116  PhysMap[0] = QCDEvMap.at(0);
117  PhysMap[21] = QCDEvMap.at(0);
118 
119  // Perform the rotation.
120  for (int i = 1; i <= 6; i++)
121  {
122  PhysMap[i] = 0;
123  PhysMap[-i] = 0;
124  for (int j = 1; j <= 6; j++)
125  {
126  PhysMap[i] += RotQCDEvToPhys[i-1][j-1] * ( QCDEvMap.at(2*j-1) + QCDEvMap.at(2*j) );
127  PhysMap[-i] += RotQCDEvToPhys[i-1][j-1] * ( QCDEvMap.at(2*j-1) - QCDEvMap.at(2*j) );
128  }
129  }
130  return PhysMap;
131  }

134  map<int,Distribution> QCDEvToPhys(map<int,Distribution> const& QCDEvMap)
135  {
136  // Fill in map in the physical basis. It attumes that the gluon
137  // has key zero and all keys from 0 to 12 exist.
138  map<int,Distribution> PhysMap;
139  PhysMap.insert({0,QCDEvMap.at(0)});
140  PhysMap.insert({0,QCDEvMap.at(0)});
141 
142  // Perform the rotation.
143  for (int i = 1; i <= 6; i++)
144  {
145  Distribution Td = RotQCDEvToPhys[i-1][0] * ( QCDEvMap.at(1) + QCDEvMap.at(2) );
146  Distribution Vd = RotQCDEvToPhys[i-1][0] * ( QCDEvMap.at(1) - QCDEvMap.at(2) );
147  for (int j = 2; j <= 6; j++)
148  {
149  Td += RotQCDEvToPhys[i-1][j-1] * ( QCDEvMap.at(2*j-1) + QCDEvMap.at(2*j) );
150  Vd += RotQCDEvToPhys[i-1][j-1] * ( QCDEvMap.at(2*j-1) - QCDEvMap.at(2*j) );
151  }
152  PhysMap.insert({i,Td});
153  PhysMap.insert({-i,Vd});
154  }
155  return PhysMap;
156  }
