18  AlphaQCD::AlphaQCD(double const& AlphaRef,
19  double const& MuRef,
20  vector<double> const& Masses,
21  vector<double> const& Thresholds,
22  int const& pt,
23  int const& nstep):
24  MatchedEvolution(AlphaRef, MuRef, Thresholds, nstep),
25  _pt(pt)
26  {
27  // Initialize all coefficients of the QCD beta function for all
28  // numbers of flavours.
29  _bQCD.resize(4,4);
30  for (int ipt = 0; ipt <= 3; ipt++)
31  for (int nf = 3; nf <= 6; nf++)
32  _bQCD(nf-3, ipt) = betaQCD(ipt, nf);
33 
34  // Compute logs of muth2 / m2 needed by the matching conditions.
35  vector<double> LogKth;
36  for (int im = 0; im < (int) Thresholds.size(); im++)
37  if (Thresholds[im] < eps12 || Masses[im] < eps12)
38  LogKth.push_back(0);
39  else
40  LogKth.push_back(2 * log( Thresholds[im] / Masses[im] ));
41 
42  // Beta function lambda function.
43  _BetaFunction = [&] (int const& nf, double const& as)-> double
44  {
45  double bt = 0, powas = as * as;
46  for (int i = 0; i <= _pt; i++)
47  {
48  bt -= powas * _bQCD(nf-3, i);
49  powas *= as;
50  }
51  return bt;
52  };
53 
54  // Matching condition lambda function.
55  _MatchingConditions = [&,LogKth] (bool const& Up, int const& nf, double const& Coup)-> double
56  {
57  const int sgn = ( Up ? 1 : -1);
58  const double ep = Coup / FourPi;
59  const double c[4] = {1, sgn * 2. / 3. * LogKth[nf], 4. / 9. * pow(LogKth[nf],2) + sgn * 38. / 3. * LogKth[nf] + sgn * 14. / 3., 0};
60  double match = 0, powep = 1;
61  for (int i = 0; i <= _pt; i++)
62  {
63  match += c[i] * powep;
64  powep *= ep;
65  }
66  return Coup * match;
67  };
68  }

89  double AlphaQCD::betaQCD(int const& pt, int const& nf) const
90  {
91  double res;
92  if ( pt == 0 )
93  res = beta0(nf);
94  else if ( pt == 1 )
95  res = beta1(nf);
96  else if ( pt == 2 )
97  res = beta2(nf);
98  else if ( pt == 3 )
99  res = beta3(nf);
100  else
101  throw runtime_error(error("AlphaQCD::betaQCD","perturbive range out-of-range."));
102 
103  return res / pow(FourPi,pt+1);
104  }
