26  map<int,DglapObjects> InitializeDglapObjectsQCD(Grid const& g,
27  vector<double> const& Masses,
28  vector<double> const& Thresholds,
29  bool const& OpEvol,
30  double const& IntEps)
31  {
32  report("Initializing DglapObjects for space-like QCD unpolarised evolution... ");
33  Timer t;
34 
35  // Compute initial and final number of active flavours according
36  // to the vector of thresholds (it assumes that the thresholds
37  // vector entries are ordered).
38  int nfi = 0;
39  int nff = Thresholds.size();
40  for (auto const& v : Thresholds)
41  if (v <= 0)
42  nfi++;
43 
44  // Compute logs of muth2 / m2 needed for the the matching
45  // conditions.
46  vector<double> LogKth;
47  for (int im = 0; im < (int) Thresholds.size(); im++)
48  if (Thresholds[im] < eps12 || Masses[im] < eps12)
49  LogKth.push_back(0);
50  else
51  LogKth.push_back(2 * log( Thresholds[im] / Masses[im] ));
52 
53  // Allocate needed operators (matching conditions and splitting
54  // functions). By now the code is fast enough to precompute
55  // everything at all available perturbative orders and the current
56  // perturbative order is accounted for only when the actual
57  // splitting functions and matching conditions (lambda) functions
58  // are defined.
59  // ===============================================================
60  // LO Matching conditions.
61  map<int,Operator> MatchLO;
62  const Operator Id {g, Identity{}, IntEps};
63  const Operator Zero{g, Null{}, IntEps};
64  MatchLO.insert({MatchingBasisQCD::PNS, Id});
65  MatchLO.insert({MatchingBasisQCD::PQQ, Id});
66  MatchLO.insert({MatchingBasisQCD::PQG, Zero});
67  MatchLO.insert({MatchingBasisQCD::PGQ, Zero});
68  MatchLO.insert({MatchingBasisQCD::PGG, Id});
69  for (int i = MatchingBasisQCD::PT3Q; i <= MatchingBasisQCD::PT35Q; i++)
70  MatchLO.insert({i, Id});
71  for (int i = MatchingBasisQCD::PT3G; i <= MatchingBasisQCD::PT35G; i++)
72  MatchLO.insert({i, Zero});
73 
74  // ===============================================================
75  // LO splitting function operators.
76  map<int,map<int,Operator>> OpMapLO;
77  const Operator O0ns{g, P0ns{}, IntEps};
78  const Operator O0qg{g, P0qg{}, IntEps};
79  const Operator O0gq{g, P0gq{}, IntEps};
80  for (int nf = nfi; nf <= nff; nf++)
81  {
82  const Operator O0gg{g, P0gg{nf}, IntEps};
83  const Operator O0qgnf = nf * O0qg;
84  map<int,Operator> OM;
85  OM.insert({EvolutionBasisQCD::PNSP, O0ns});
86  OM.insert({EvolutionBasisQCD::PNSM, O0ns});
87  OM.insert({EvolutionBasisQCD::PNSV, O0ns});
88  OM.insert({EvolutionBasisQCD::PQQ, O0ns});
89  OM.insert({EvolutionBasisQCD::PQG, O0qgnf});
90  OM.insert({EvolutionBasisQCD::PGQ, O0gq});
91  OM.insert({EvolutionBasisQCD::PGG, O0gg});
92  OpMapLO.insert({nf, OM});
93  }
94 
95  // ===============================================================
96  // NLO Matching conditions.
97  map<int,map<int,Operator>> MatchNLO;
98  const Operator AS1HgL {g, AS1Hg_L{}, IntEps};
99  const Operator AS1ggHL{g, AS1ggH_L{}, IntEps};
100  for (int nf = nfi; nf <= nff; nf++)
101  {
102  const Operator AS1Hg = LogKth[nf] * AS1HgL;
103  const Operator AS1ggH = LogKth[nf] * AS1ggHL;
104  const Operator AS1Tg = - nf * AS1Hg;
105  map<int,Operator> OM;
106  OM.insert({MatchingBasisQCD::PNS, Zero});
107  OM.insert({MatchingBasisQCD::PQQ, Zero});
108  OM.insert({MatchingBasisQCD::PQG, AS1Hg});
109  OM.insert({MatchingBasisQCD::PGQ, Zero});
110  OM.insert({MatchingBasisQCD::PGG, AS1ggH});
111  for (int i = MatchingBasisQCD::PT3Q; i <= MatchingBasisQCD::PT35Q; i++)
112  OM.insert({i, Zero});
113  for (int i = MatchingBasisQCD::PT3G; i <= MatchingBasisQCD::PT35G; i++)
114  if (i > MatchingBasisQCD::PT3G + nf - 1)
115  OM.insert({i, AS1Hg});
116  else if (i == MatchingBasisQCD::PT3G + nf - 1)
117  OM.insert({i, AS1Tg});
118  else
119  OM.insert({i, Zero});
120  MatchNLO.insert({nf, OM});
121  }
122 
123  // ===============================================================
124  // NLO splitting function operators.
125  map<int,map<int,Operator>> OpMapNLO;
126  for (int nf = nfi; nf <= nff; nf++)
127  {
128  const Operator O1nsp{g, P1nsp{nf}, IntEps};
129  const Operator O1nsm{g, P1nsm{nf}, IntEps};
130  const Operator O1ps {g, P1ps{nf}, IntEps};
131  const Operator O1qg {g, P1qg{nf}, IntEps};
132  const Operator O1gq {g, P1gq{nf}, IntEps};
133  const Operator O1gg {g, P1gg{nf}, IntEps};
134  const Operator O1qq = O1nsp + O1ps;
135  map<int,Operator> OM;
136  OM.insert({EvolutionBasisQCD::PNSP, O1nsp});
137  OM.insert({EvolutionBasisQCD::PNSM, O1nsm});
138  OM.insert({EvolutionBasisQCD::PNSV, O1nsm});
139  OM.insert({EvolutionBasisQCD::PQQ, O1qq});
140  OM.insert({EvolutionBasisQCD::PQG, O1qg});
141  OM.insert({EvolutionBasisQCD::PGQ, O1gq});
142  OM.insert({EvolutionBasisQCD::PGG, O1gg});
143  OpMapNLO.insert({nf, OM});
144  }
145 
146  // ===============================================================
147  // NNLO Matching conditions.
148  map<int,map<int,Operator>> MatchNNLO;
149  const Operator APS2Hq0 {g, APS2Hq_0{}, IntEps};
150  const Operator APS2HqL {g, APS2Hq_L{}, IntEps};
151  const Operator APS2HqL2 {g, APS2Hq_L2{}, IntEps};
152  const Operator ANS2qqH0 {g, ANS2qqH_0{}, IntEps};
153  const Operator ANS2qqHL {g, ANS2qqH_L{}, IntEps};
154  const Operator ANS2qqHL2{g, ANS2qqH_L2{}, IntEps};
155  const Operator AS2Hg0 {g, AS2Hg_0{}, IntEps};
156  const Operator AS2HgL {g, AS2Hg_L{}, IntEps};
157  const Operator AS2HgL2 {g, AS2Hg_L2{}, IntEps};
158  const Operator AS2gqH0 {g, AS2gqH_0{}, IntEps};
159  const Operator AS2gqHL {g, AS2gqH_L{}, IntEps};
160  const Operator AS2gqHL2 {g, AS2gqH_L2{}, IntEps};
161  const Operator AS2ggH0 {g, AS2ggH_0{}, IntEps};
162  const Operator AS2ggHL {g, AS2ggH_L{}, IntEps};
163  const Operator AS2ggHL2 {g, AS2ggH_L2{}, IntEps};
164  const Operator AS2qqH0 = ANS2qqH0 + APS2Hq0;
165  const Operator AS2qqHL = ANS2qqHL + APS2HqL;
166  const Operator AS2qqHL2 = ANS2qqHL2 + APS2HqL2;
167  for (int nf = nfi; nf <= nff; nf++)
168  {
169  const double lnk = LogKth[nf];
170  const double lnk2 = lnk * lnk;
171  const Operator APS2Hq = APS2Hq0 + lnk * APS2HqL + lnk2 * APS2HqL2;
172  const Operator ANS2qqH = ANS2qqH0 + lnk * ANS2qqHL + lnk2 * ANS2qqHL2;
173  const Operator AS2Hg = AS2Hg0 + lnk * AS2HgL + lnk2 * AS2HgL2;
174  const Operator AS2gqH = AS2gqH0 + lnk * AS2gqHL + lnk2 * AS2gqHL2;
175  const Operator AS2ggH = AS2ggH0 + lnk * AS2ggHL + lnk2 * AS2ggHL2;
176  const Operator AS2qqH = AS2qqH0 + lnk * AS2qqHL + lnk2 * AS2qqHL2;
177  const Operator AS2TqH = ANS2qqH - nf * APS2Hq;
178  const Operator AS2Tg = - nf * AS2Hg;
179  map<int,Operator> OM;
180  OM.insert({MatchingBasisQCD::PNS, ANS2qqH});
181  OM.insert({MatchingBasisQCD::PQQ, AS2qqH});
182  OM.insert({MatchingBasisQCD::PQG, AS2Hg});
183  OM.insert({MatchingBasisQCD::PGQ, AS2gqH});
184  OM.insert({MatchingBasisQCD::PGG, AS2ggH});
185  for (int i = MatchingBasisQCD::PT3Q; i <= MatchingBasisQCD::PT35Q; i++)
186  if (i > MatchingBasisQCD::PT3Q + nf - 1)
187  OM.insert({i, AS2qqH});
188  else if (i == MatchingBasisQCD::PT3Q + nf - 1)
189  OM.insert({i, AS2TqH});
190  else
191  OM.insert({i, ANS2qqH});
192  for (int i = MatchingBasisQCD::PT3G; i <= MatchingBasisQCD::PT35G; i++)
193  if (i > MatchingBasisQCD::PT3G + nf - 1)
194  OM.insert({i, AS2Hg});
195  else if (i == MatchingBasisQCD::PT3G + nf - 1)
196  OM.insert({i, AS2Tg});
197  else
198  OM.insert({i, Zero});
199  MatchNNLO.insert({nf, OM});
200  }
201 
202  // ===============================================================
203  // NNLO splitting function operators.
204  map<int,map<int,Operator>> OpMapNNLO;
205  for (int nf = nfi; nf <= nff; nf++)
206  {
207  const Operator O2nsp{g, P2nsp{nf}, IntEps};
208  const Operator O2nsm{g, P2nsm{nf}, IntEps};
209  const Operator O2nss{g, P2nss{nf}, IntEps};
210  const Operator O2ps {g, P2ps{nf}, IntEps};
211  const Operator O2qg {g, P2qg{nf}, IntEps};
212  const Operator O2gq {g, P2gq{nf}, IntEps};
213  const Operator O2gg {g, P2gg{nf}, IntEps};
214  const Operator O2qq = O2nsp + O2ps;
215  const Operator O2nsv = O2nsm + O2nss;
216  map<int,Operator> OM;
217  OM.insert({EvolutionBasisQCD::PNSP, O2nsp});
218  OM.insert({EvolutionBasisQCD::PNSM, O2nsm});
219  OM.insert({EvolutionBasisQCD::PNSV, O2nsv});
220  OM.insert({EvolutionBasisQCD::PQQ, O2qq});
221  OM.insert({EvolutionBasisQCD::PQG, O2qg});
222  OM.insert({EvolutionBasisQCD::PGQ, O2gq});
223  OM.insert({EvolutionBasisQCD::PGG, O2gg});
224  OpMapNNLO.insert({nf, OM});
225  }
226 
227  // ===============================================================
228  // NNNLO splitting function operators. For now only the
229  // non-singlet splitting functions have been computed to leading
230  // colour even though the subleading colour part is estimated
231  // through an approximate parameterisation.
232  map<int,map<int,Operator>> OpMapNNNLO;
233  for (int nf = nfi; nf <= nff; nf++)
234  {
235  const Operator O3nsp{g, P3nsp{nf}, IntEps};
236  const Operator O3nsm{g, P3nsm{nf}, IntEps};
237  const Operator O3nss{g, P3nss{nf}, IntEps};
238  const Operator O3nsv = O3nsm + O3nss;
239  map<int,Operator> OM;
240  OM.insert({EvolutionBasisQCD::PNSP, O3nsp});
241  OM.insert({EvolutionBasisQCD::PNSM, O3nsm});
242  OM.insert({EvolutionBasisQCD::PNSV, O3nsv});
243  OM.insert({EvolutionBasisQCD::PQQ, Zero});
244  OM.insert({EvolutionBasisQCD::PQG, Zero});
245  OM.insert({EvolutionBasisQCD::PGQ, Zero});
246  OM.insert({EvolutionBasisQCD::PGG, Zero});
247  OpMapNNNLO.insert({nf, OM});
248  }
249 
250  // Define object of the structure containing the DglapObjects.
251  map<int,DglapObjects> DglapObj;
252 
253  // Allocate convolution maps for evolution and matching, and set
254  // of operators.
255  for (int nf = nfi; nf <= nff; nf++)
256  {
257  DglapObjects obj;
258  obj.Threshold = Thresholds[nf-1];
259  if (OpEvol)
260  {
261  const EvolutionOperatorBasisQCD evopb{nf};
262  const MatchingOperatorBasisQCD mtopb{nf};
263  obj.SplittingFunctions.insert({0, Set<Operator>{evopb, OpMapLO.at(nf)}});
264  obj.SplittingFunctions.insert({1, Set<Operator>{evopb, OpMapNLO.at(nf)}});
265  obj.SplittingFunctions.insert({2, Set<Operator>{evopb, OpMapNNLO.at(nf)}});
266  obj.SplittingFunctions.insert({3, Set<Operator>{evopb, OpMapNNNLO.at(nf)}});
267  obj.MatchingConditions.insert({0, Set<Operator>{mtopb, MatchLO}});
268  obj.MatchingConditions.insert({1, Set<Operator>{mtopb, MatchNLO.at(nf)}});
269  obj.MatchingConditions.insert({2, Set<Operator>{mtopb, MatchNNLO.at(nf)}});
270  }
271  else
272  {
273  const EvolutionBasisQCD evb{nf};
274  const MatchingBasisQCD mtb{nf};
275  obj.SplittingFunctions.insert({0, Set<Operator>{evb, OpMapLO.at(nf)}});
276  obj.SplittingFunctions.insert({1, Set<Operator>{evb, OpMapNLO.at(nf)}});
277  obj.SplittingFunctions.insert({2, Set<Operator>{evb, OpMapNNLO.at(nf)}});
278  obj.SplittingFunctions.insert({3, Set<Operator>{evb, OpMapNNNLO.at(nf)}});
279  obj.MatchingConditions.insert({0, Set<Operator>{mtb, MatchLO}});
280  obj.MatchingConditions.insert({1, Set<Operator>{mtb, MatchNLO.at(nf)}});
281  obj.MatchingConditions.insert({2, Set<Operator>{mtb, MatchNNLO.at(nf)}});
282  }
283  DglapObj.insert({nf,obj});
284  }
285  t.stop();
286 
287  return DglapObj;
288  }

772  function<Set<Operator>(int const&, double const&)> SplittingFunctions(map<int,DglapObjects> const& DglapObj,
773  int const& PerturbativeOrder,
774  function<double(double const&)> const& Alphas)
775  {
776  if (PerturbativeOrder == 0)
777  return [=] (int const& nf, double const& mu) -> Set<Operator>
778  {
779  const double cp = Alphas(mu) / FourPi;
780  return cp * DglapObj.at(nf).SplittingFunctions.at(0);
781  };
782  else if (PerturbativeOrder == 1)
783  return [=] (int const& nf, double const& mu) -> Set<Operator>
784  {
785  const double cp = Alphas(mu) / FourPi;
786  const auto sf = DglapObj.at(nf).SplittingFunctions;
787  return cp * ( sf.at(0) + cp * sf.at(1) );
788  };
789  else if (PerturbativeOrder == 2)
790  return [=] (int const& nf, double const& mu) -> Set<Operator>
791  {
792  const double cp = Alphas(mu) / FourPi;
793  const auto sf = DglapObj.at(nf).SplittingFunctions;
794  return cp * ( sf.at(0) + cp * ( sf.at(1) + cp * sf.at(2) ) );
795  };
796  else if (PerturbativeOrder == 3)
797  return [=] (int const& nf, double const& mu) -> Set<Operator>
798  {
799  const double cp = Alphas(mu) / FourPi;
800  const auto sf = DglapObj.at(nf).SplittingFunctions;
801  return cp * ( sf.at(0) + cp * ( sf.at(1) + cp * ( sf.at(2) + cp * sf.at(3) ) ) );
802  };
803  else
804  throw runtime_error(error("SplittingFunctions","Perturbative order not allowed."));
805  }

807  //_____________________________________________________________________________
808  function<Set<Operator>(bool const&, int const&)> MatchingConditions(map<int,DglapObjects> const& DglapObj,
809  int const& PerturbativeOrder,
810  function<double(double const&)> const& Alphas)
811  {
812  // Compute coupling above and below the thresholds.
813  map<int,double> asThUp;
814  map<int,double> asThDown;
815  for (auto obj = std::next(DglapObj.begin()); obj != DglapObj.end(); ++obj)
816  {
817  const int nf = obj->first;
818  const double thr = obj->second.Threshold;
819  asThDown.insert({nf,Alphas(thr*(1-eps8))/FourPi});
820  asThUp.insert({nf,Alphas(thr*(1+eps8))/FourPi});
821  }
822 
823  if (PerturbativeOrder == 0)
824  return [=] (bool const&, int const& nf) -> Set<Operator>
825  {
826  return DglapObj.at(nf).MatchingConditions.at(0);
827  };
828  else if (PerturbativeOrder == 1)
829  return [=] (bool const& Up, int const& nf) -> Set<Operator>
830  {
831  const double cp = ( Up ? asThUp.at(nf+1) : asThDown.at(nf+1) );
832  const auto mc = DglapObj.at(nf).MatchingConditions;
833  return mc.at(0) + ( Up ? 1 : -1) * cp * mc.at(1);
834  };
835  else if (PerturbativeOrder == 2)
836  return [=] (bool const& Up, int const& nf) -> Set<Operator>
837  {
838  const double cp = ( Up ? asThUp.at(nf+1) : asThDown.at(nf+1) );
839  const auto mc = DglapObj.at(nf).MatchingConditions;
840  return mc.at(0) + ( Up ? 1 : -1) * cp * ( mc.at(1) + cp * mc.at(2) );
841  };
842  else if (PerturbativeOrder == 3)
843  return [=] (bool const& Up, int const& nf) -> Set<Operator>
844  {
845  const double cp = ( Up ? asThUp.at(nf+1) : asThDown.at(nf+1) );
846  const auto mc = DglapObj.at(nf).MatchingConditions;
847  return mc.at(0) + ( Up ? 1 : -1) * cp * ( mc.at(1) + cp * mc.at(2) );
848  };
849  else
850  throw runtime_error(error("MatchingConditions","Perturbative order not allowed."));
851  }
