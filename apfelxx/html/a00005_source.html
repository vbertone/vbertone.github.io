22  double Cm21gNC::Regular(double const& x) const
23  {
24  /*
25  const double eta = this->_eta;
26  const double z = eta * x;
27  double xi = 4 * eta / ( 1 - eta );
28  double wr = xi * ( 1 / z - 1 ) / 4 - 1;
29  const double cm21g = xi * c2log_(&wr,&xi) / z / M_PI;
30  return cm21g;
31  */
32  const double eta = this->_eta;
33  const double z = eta * x;
34  const double z2 = z * z;
35  const double epsi = ( 1 - eta ) / eta / 4;
36  const double epsi2 = epsi * epsi;
37  const double v = sqrt( 1 - 4 * epsi * z / ( 1 - z ));
38  const double cm21g = 4 * TR * ( ( z2 + ( 1 - z ) * ( 1 - z ) + 4 * epsi * z * ( 1 - 3 * z )
39  - 8 * epsi2 * z2 ) * log( ( 1 + v ) / ( 1 - v ) )
40  + ( 8 * z * ( 1 - z ) - 1 - 4 * epsi * z * ( 1 - z ) ) * v );
41  return cm21g;
42  }

71  Expression(eta)

73  // Compute integral needed to enforce the Adler sum rule. See
74  // eqs. (71) and (97) of https://arxiv.org/pdf/1001.2312.pdf.
75  const Integrator Integrand{[&] (double const& y)->double{ return Regular(y); }};
76  _adler = - Integrand.integrate(0,1,eps5);
77  }
78  double Cm22nsNC::Regular(double const& x) const
79  {
80  /*
81  const double eta = this->_eta;
82  const double z = eta * x;
83  double xi = 4 * eta / ( 1 - eta );
84  double wr = xi * ( 1 / z - 1 ) / 4 - 1;
85  const double cm22ns = 16 * M_PI * xi * d2nloq_(&wr,&xi) / z;
86  return cm22ns;
87  */
88  const double eta = this->_eta;
89  const double xi = 4 * eta / ( 1 - eta );
90  const double z = eta * x;
91  const double z2 = z * z;
92  const double z3 = z * z2;
93  const double omz = 1 - z;
94  const double zr2 = z2 / xi / xi;
95  const double qr = z / omz / xi;
96  const double qr2 = qr * qr;
97  const double sq1 = sqrt( 1 - 4 * z / xi / omz );
98  const double sq2 = sqrt( 1 - 4 * z / xi );
99  if (sq1 == sq2)
100  return 0;
101  const double l1 = log( ( 1 + sq1 ) / ( 1 - sq1 ) );
102  const double l2 = log( ( 1 + sq2 ) / ( 1 - sq2 ) );
103  const double l3 = log( ( sq2 + sq1 ) / ( sq2 - sq1 ) );
104  const double dil1 = dilog( omz * ( 1 + sq1 ) / ( 1 + sq2 ) );
105  const double dil2 = dilog( ( 1 - sq2 ) / ( 1 + sq1 ) );
106  const double dil3 = dilog( ( 1 - sq1 ) / ( 1 + sq2 ) );
107  const double dil4 = dilog( ( 1 + sq1 ) / ( 1 + sq2 ) );
108  const double cm22ns = CF * TR * ( ( 4. / 3 * ( 1 + z2 ) / omz - 16 / omz * zr2 * ( 1 - 9 * z + 9 * z2 ) )
109  * ( log( omz / z2 ) * l1 + l1 * l2 + 2 * ( - dil1 + dil2 + dil3 - dil4 ) )
110  + ( - 8. / 3 + 4 / omz + qr2 * ( 128. - 432 * z + 288 * z2 - 8 / omz ) ) * l1
111  + ( 88. / 9 + 136. / 9 * z - 152. / 9 / omz
112  + qr * ( 464. / 9 - 512. / 3 * z + 2048. / 9 * z2 )
113  + qr2 * ( - 832. / 9 + 6208. / 9 * z - 11392. / 9 * z2 + 6016. / 9 * z3 ) ) * l3 / sq2
114  + ( - 272. / 27 - 1244. / 27 * z + 718. / 27 / omz
115  + qr * ( - 3424. / 27 + 15608. / 27 * z - 4304. / 9 * z2 + 20. / 27 / omz ) ) * sq1 );
116  return cm22ns;
117  }

142  const double omz = 1 - z;
143  const double zr2 = z2 / xi / xi;
144  const double qr = z / omz / xi;
145  const double qr2 = qr * qr;
146  const double sq1 = sqrt( 1 - 4 * z / xi / omz );
147  const double sq2 = sqrt( 1 - 4 * z / xi );
148  if (sq1 == sq2)

153  const double dil1 = dilog( omz * ( 1 + sq1 ) / ( 1 + sq2 ) );
154  const double dil2 = dilog( ( 1 - sq2 ) / ( 1 + sq1 ) );
155  const double dil3 = dilog( ( 1 - sq1 ) / ( 1 + sq2 ) );
156  const double dil4 = dilog( ( 1 + sq1 ) / ( 1 + sq2 ) );
157  const double cml2ns = CF * TR * ( 96 * z * zr2 * ( log( omz / z2 ) * l1 + l1 * l2 + 2 * ( - dil1 + dil2 + dil3 - dil4 ) )
158  + qr2 * ( 64 - 288 * z + 192 * z2 ) * l1
159  + z * ( 16. / 3 - 416 * z / 3 / xi + 1408 * z2 / 3 / xi / xi ) * l3 / sq2
160  + ( 16. / 3 - 400 * z / 18
