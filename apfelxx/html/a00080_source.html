20  template<class T>
21  QGrid<T>::QGrid(int const& nQ,
22  double const& QMin,
23  double const& QMax,
24  int const& InterDegree,
25  vector<double> const& Thresholds,
26  function<double(double const&)> const& TabFunc,
27  function<double(double const&)> const& InvTabFunc):
28  _nQ(nQ),
29  _QMin(QMin),
30  _QMax(QMax),
31  _InterDegree(InterDegree),
32  _Thresholds(Thresholds),
33  _TabFunc(TabFunc)
34  {
35  // Check that QMin is actually smaller than QMax.
36  if (QMax <= QMin)
37  throw runtime_error(error("QGrid::QGrid","QMax must be larger than QMin"));
38 
39  // Check that "TabFunc" and "InvTabFunc" are actually the inverse
40  // function of each other. The check is done at grid bounds and in
41  // the middle.
42 
43  const vector<double> TestPoints{_QMin, ( _QMax + _QMin ) / 2, _QMax};
44  for (auto const& p : TestPoints)
45  {
46  const double reldiff = abs(InvTabFunc( TabFunc(p) ) / p - 1);
47  if (reldiff > eps8)
48  throw runtime_error(error("QGrid::QGrid","TabFunc and InvTabFunc are not the inverse of each other."));
49  }
50 
51  // Find initial and final number of flavours.
52  const int nfin = NF(_QMin, _Thresholds);
53  const int nffi = NF(_QMax, _Thresholds);
54 
55  // Compute a temporary grid constant in ln(ln(Q^2/Lambda^2)
56  // without taking into account the threholds.
57  vector<double> fq = {_TabFunc(_QMin)};
58  double Step = ( _TabFunc(_QMax) - _TabFunc(_QMin) ) / _nQ;
59  for (int iq = 1; iq <= _nQ; iq++)
60  fq.push_back(fq.back()+Step);
61 
62  // Identify the indices of the grid points that fall right below
63  // the thresholds. This will define the number of points that
64  // fall in each interval defined by the grid bounds and the
65  // thresholds.
66  _nQg.push_back(0);
67  vector<double> fqTh = {_TabFunc(_QMin)};
68  for (int isg = nfin+1; isg <= nffi; isg++)
69  {
70  fqTh.push_back(_TabFunc(_Thresholds[isg-1]));
71  _nQg.push_back(lower_bound(fq.begin()+1, fq.end(), fqTh.back()) - fq.begin());
72  }
73  _nQg.push_back(_nQ);
74  fqTh.push_back(_TabFunc(_QMax));
75 
76  // Check that all intervals have at least two points. If not,
77  // adjust it. Check also whether there is any subgrid whose number
78  // of points is smaller than the interpolation degree plus one.
79  // If so, adjust the interpolation degree.
80  for (int isg = 0; isg < (int) _nQg.size() - 1; isg++)
81  {
82  if (_nQg[isg+1] - _nQg[isg] < 2)
83  _nQg[isg+1] = _nQg[isg] + 2;
84  if (_nQg[isg+1] - _nQg[isg] < _InterDegree + 2)
85  _InterDegree = _nQg[isg+1] - _nQg[isg] - 1;
86  }
87 
88  // Adjust _nQ if needed.
89  if (_nQ != _nQg.back()) _nQ = _nQg.back();
90 
91  // Now construct the actual grid in such a way that the threshold
92  // concides with nodes of the grid.
93  _fQg.push_back(_TabFunc(_QMin));
94  for (int isg = 0; isg < (int) _nQg.size() - 1; isg++)
95  {
96  Step = ( fqTh[isg+1] - fqTh[isg] ) / ( _nQg[isg+1] - _nQg[isg] - 1 );
97  for (int iq = _nQg[isg] + 1; iq < _nQg[isg+1]; iq++) _fQg.push_back(_fQg.back()+Step);
98  _fQg.push_back(_fQg.back());
99  }
100 
101  // Now compute grid in Q.
102  for (auto const& lq : _fQg)
103  _Qg.push_back(InvTabFunc(lq));
104 
105  // Displace slightly the values below and above the thresholds.
106  for (int isg = 1; isg < (int) _nQg.size() - 1; isg++)
107  {
108  _Qg[_nQg[isg]-1] *= 1 - eps12;
109  _Qg[_nQg[isg]] *= 1 + eps12;
110  _fQg[_nQg[isg]-1] = TabFunc(_Qg[_nQg[isg]-1]);
111  _fQg[_nQg[isg]] = TabFunc(_Qg[_nQg[isg]]);
112  }
113  }

130  template<class T>
131  double QGrid<T>::Interpolant(int const& tQ, int const& tau, double const& fq) const
132  {
133  // Return immediately 1 if "Q" coincides with "_Qg[tau]" unless
134  // tau coincides with a threshold index. In that case return zero.
135  if (abs(fq / _fQg[tau] - 1) < eps11)
136  return 1;
137 
138  // Define the lower bound of the interpolation range.
139  int bound = tau + tQ - _InterDegree;
140  if (_InterDegree > tau + tQ)
141  bound = 0;
142  //if (fq < _fQg[bound] || fq >= _fQg[tau+tQ+1])
143  // return 0;
144 
145  // Initialize interpolant
146  double w_int = 1;
147 
148  // Find the the neighbors of "Q" on the grid
149  int j;
150  for (j = tau+tQ-bound; j >=0; j--)
151  if (fq < _fQg[tau+tQ-j+1])
152  break;
153 
154  // Compute the interpolant
155  for (int delta = tau-j; delta <= tau-j+_InterDegree; delta++)
156  if (delta != tau)
157  w_int *= ( fq - _fQg[delta] ) / ( _fQg[tau] - _fQg[delta] );
158 
159  return w_int;
160  }

163  template<class T>
164  tuple<int,int,int> QGrid<T>::SumBounds(double const& Q) const
165  {
166  tuple<int,int,int> bounds{0, 0, 0};
167 
168  // Return if "Q" is outside the grid range (no sum will be
169  // performed).
170  if (Q < _Qg.front() - eps12 || Q > _Qg.back() + eps12)
171  return bounds;
172 
173  // If Q falls in the tiny gap at the thresholds assume the node
174  // below.
175  for (int iQ = 1; iQ < (int) _nQg.size() - 1; iQ++)
176  if (Q > _Qg[_nQg[iQ]-1] && Q <= _Qg[_nQg[iQ]])
177  {
178  get<1>(bounds) = _nQg[iQ] - 1;
179  get<2>(bounds) = _nQg[iQ];
180  return bounds;
181  }
182 
183  // Identify the subgrid in which Q falls.
184  int iQ;
185  for (iQ = 0; iQ < (int) _nQg.size() - 1; iQ++)
186  if (Q > _Qg[_nQg[iQ]] && Q <= _Qg[_nQg[iQ+1]])
187  break;
188 
189  if (iQ == (int) _nQg.size()-1)
190  iQ--;
191 
192  // Determine the control parameter and put it in the first entry
193  // of the tuple.
194  if (Q > _Qg[_nQg[iQ+1] - _InterDegree])
195  {
196  int id;
197  for (id = 2; id <= _InterDegree; id++)
198  if (Q > _Qg[_nQg[iQ+1] - id])
199  break;
200  get<0>(bounds) = _InterDegree - id + 1;
201  }
202 
203  // Determine the actual bounds.
204  const int low = lower_bound(_Qg.begin()+1, _Qg.end(), Q) - _Qg.begin();
205  get<1>(bounds) = low;
206  get<2>(bounds) = low;
207 
208  if (abs(Q / _Qg[low] - 1) <= eps12)
209  get<2>(bounds) += 1;
210  else
211  {
212  get<1>(bounds) += - get<0>(bounds) - 1;
213  get<2>(bounds) += - get<0>(bounds) + _InterDegree;
214  }
215 
216  return bounds;
217  }

220  template<class T>
221  T QGrid<T>::Evaluate(double const& Q) const
222  {
223  const tuple<int,int,int> bounds = SumBounds(Q);
224  const double fq = _TabFunc(Q);
225 
226  // first create a copy of template object with the first component
227  int tau = get<1>(bounds);
228  T result = Interpolant(get<0>(bounds), tau, fq) * _GridValues[tau];
229 
230  // then loop and add the extra terms
231  for (tau = tau+1; tau < get<2>(bounds); tau++)
232  result += Interpolant(get<0>(bounds), tau, fq) * _GridValues[tau];
233 
234  return result;
235  }
