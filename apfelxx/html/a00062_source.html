22  double LagrangeInterpolator::Interpolant(int const& beta, double const& lnx, SubGrid const& sg) const
23  {
24  // Get the logarithmic grid.
25  auto const& lxsg = sg.GetLogGrid();
26 
27  // Return immediately 1 if "x" coincides with "xg[beta]".
28  if (abs(lnx - lxsg[beta]) < eps12)
29  return 1;
30 
31  // Return 0 if "x" is outside the range in which the interpolant
32  // is different from zero. Ideally this functions should never be
33  // called if "beta" and "x" are such that "Interpolant" is
34  // identically zero. Use "SumBounds" to know where "beta" should
35  // run over given "x".
36  const int id = sg.InterDegree();
37  int bound = beta - id;
38  if (id > beta)
39  bound = 0;
40  if (lnx < lxsg[bound] || lnx >= lxsg[beta+1])
41  return 0;
42 
43  // Find the the neighbors of "x" on the grid.
44  int j;
45  for (j = 0; j <= beta-bound; j++)
46  if (lnx >= lxsg[beta-j])
47  break;
48 
49  // Compute the interpolant.
50  double w_int = 1;
51  for (int delta = beta-j; delta <= beta-j+id; delta++)
52  if (delta != beta)
53  w_int *= ( lnx - lxsg[delta] ) / ( lxsg[beta] - lxsg[delta] );
54 
55  return w_int;
56  }

59  array<int, 2> LagrangeInterpolator::SumBounds(double const& x, SubGrid const& sg) const
60  {
61  auto const& xsg = sg.GetGrid();
62 
63  array<int,2> bounds = {{0, 0}};
64  if (x < xsg[0] - eps12 || x > xsg[sg.nx()] + eps12)
65  return bounds;
66 
67  const int low = lower_bound(xsg.begin()+1, xsg.end()-sg.InterDegree()-1, x) - xsg.begin();
68  bounds[0] = bounds[1] = low;
69 
70  if (abs(x - xsg[low]) <= eps12)
71  bounds[1] += 1;
72  else
73  {
74  bounds[0] -= 1;
75  bounds[1] += sg.InterDegree();
76  }
77 
78  return bounds;
79  }
