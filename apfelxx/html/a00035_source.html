91  const double pqq = 2 / ( 1 - x ) - 1 - x;
92  const double pqqmx = 2 / ( 1 + x ) - 1 + x;
93  const double S2x = - 2 * dilog(-x) + lnx2 / 2 - 2 * lnx * log(1+x) - Pi2 / 6;
94  const double gqq1 =
95  + 2 * CF * _nf * ( ( - 10 / 9. - 2 * lnx / 3 ) * pqq - 4 * ( 1 - x ) / 3 )
96  + 4 * CA * CF * ( ( 67 / 18. + 11 * lnx / 6 + lnx2 / 2 - Pi2 / 6 ) * pqq

98  + 4 * CF * CF * ( ( 3 * lnx / 2 + 2 * ln1mx * lnx - 2 * lnx2 ) * pqq - 5 * ( 1 - x )
99  + lnx2 * ( 1 + x ) / 2 - lnx * ( 3.5 + 3 * x / 2 ) )
100  + 4 * CF * ( CF - CA / 2 ) * ( 2 * pqqmx * S2x + 4 * ( 1 - x ) + 2 * lnx * ( 1 + x ) );
101  const double gqq1l = _a2 / ( 1 - x );
102  const double x1nspa = gqq1 - gqq1l;
