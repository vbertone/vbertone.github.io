47  if (x == 1) {
48  H = PI6;
49  } else if (x == -1) {
50  H = -PI12;
51  } else {

54  Y = -1/(1+T);

59  } else if (T < -1) {

74  } else if (T <= 1) {

76  S = 1;

78  } else {

83  }

86  B1 = 0;
87  B2 = 0;
88  for (int i=19;i>=0;i--){
89  B0 = C[i] + ALFA*B1-B2;
90  B2 = B1;
91  B1 = B0;
92  }
93  H = -(S*(B0-H*B2)+A);
94  }
95  return H;
