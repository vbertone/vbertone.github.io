298  map<int,Distribution> DistributionMap(Grid const& g,
299  function<map<int,double>(double const&, double const&)> const& InDistFunc,
300  double const& Q,
301  vector<int> const& skip)
302  {
303  // Joint grid and subgrid vectors.
304  const vector<double> jg = g.GetJointGrid().GetGrid();
305 
306  // Initialise output.
307  map<int,Distribution> DistMap;
308  const map<int,double> f = InDistFunc(jg[0],Q);
309  for (auto it = f.begin(); it != f.end(); ++it)
310  if (find(skip.begin(), skip.end(), it->first) == skip.end())
311  DistMap.insert({it->first,Distribution{g}});
312 
313  // Fill in joint grid.
314  for (auto const& ix : jg)
315  {
316  const map<int,double> f = InDistFunc(ix < 1 ? ix : 1,Q);
317  for (auto it = f.begin(); it != f.end(); ++it)
318  if (find(skip.begin(), skip.end(), it->first) == skip.end())
319  DistMap.at(it->first).PushJointGrid(it->second);
320  }
321 
322  // Fill in subgrids.
323  for (int ig = 0; ig < g.nGrids(); ig++)
324  {
325  bool next = true;
326  for (auto const& ix: g.GetSubGrid(ig).GetGrid())
327  {
328  const map<int,double> f = InDistFunc(ix < 1 ? ix : 1,Q);
329  for (auto it = f.begin(); it != f.end(); ++it)
330  if (find(skip.begin(), skip.end(), it->first) == skip.end())
331  DistMap.at(it->first).PushSubGrid(it->second, next);
332  next = false;
333  }
334  }
335 
336  return DistMap;
337  }
