86  double Integrator::integrate(double const& a, double const& b, double const& eps) const
87  {
88  const double delta = eps25 * abs(a-b);
89  double dgauss = 0;
90  double aa = a;
91 
92  goto5:
93  double y = b - aa;
94  if (abs(y) <= delta)
95  return dgauss;
96 
97  goto2:
98  double bb = aa + y;
99  double c1 = 0.5 * ( aa + bb );
100  double c2 = c1 - aa;
101 
102  double s8 = 0;
103  for (int i = 0; i < 4; i++)
104  {
105  double u = gq_x[1][i] * c2;
106  s8 += gq_w[1][i] * ( integrand(c1+u) + integrand(c1-u) );
107  }
108 
109  double s16 = 0;
110  for (int i = 0; i < 8; i++)
111  {
112  double u = gq_x[2][i] * c2;
113  s16 += gq_w[2][i] * ( integrand(c1+u) + integrand(c1-u) );
114  }
115 
116  s8 *= c2;
117  s16 *= c2;
118  if (abs(s16-s8) > eps*(1+abs(s16)))
119  goto goto4;
120 
121  dgauss += s16;
122  aa = bb;
123  goto goto5;
124 
125  goto4:
126  y = 0.5 * y;
127  if (abs(y) > delta)
128  goto goto2;
129 
130  throw runtime_error(error("Integrator::dgauss", "too high accuracy required"));
131 
132  return dgauss;
133  }

136  double Integrator::integrate(double const& a, double const& b, vector<double> const& FixPts, double const& eps) const
137  {
138  // Create vector of fixed points including the integration bounds
139  // in ordered sequence and removing the points outside the
140  // integration range.
141  vector<double> bounds{(a < b ? a : b), (a < b ? b : a)};
142  for (auto const& ib : FixPts)
143  if(ib > bounds.front() && ib < bounds.back())
144  bounds.push_back(ib);
145 
146  // Sort vector according on how the integration bounds are
147  // ordered.
148  const auto sorter = [a,b] (double const& x1, double const& x2)->bool
149  {
150  if (b > a)
151  return x1 < x2;
152  else
153  return x1 > x2;
154  };
155  sort(bounds.begin(),bounds.end(), sorter);
156 
157  // Now compute sub-integrals and sum them.
158  double dgauss = 0;
159  for (int i = 1; i < (int) bounds.size(); i++)
160  dgauss += integrate(bounds[i-1],bounds[i],eps);
161 
162  return dgauss;
163  }
