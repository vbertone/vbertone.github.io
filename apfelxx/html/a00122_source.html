20  function<double(double const&)> TmdCrossSectionDY(double const& Vs,
21  double const& Q,
22  double const& y,
23  function<Set<Distribution>(double const&, double const&, double const&)> const& EvolvedTMDPDFs,
24  function<double(double const&)> const& Alphas,
25  function<vector<double>(double const&)> const& fEWCharges,
26  int const& PerturbativeOrder,
27  vector<double> const& Thresholds,
28  double const& cmuf,
29  double const& czetaf)
30  {
31  // Compute "muf" and "zetaf". They are assumed to be proportional
32  // to Q and Q^2, respectively, through "cmuf" and "czetaf".
33  const double muf = cmuf * Q;
34  const double zetaf = czetaf * Q * Q;
35 
36  // Compute values of x1 and x2.
37  const double x1 = Q * exp(-y) / Vs;
38  const double x2 = Q * exp(y) / Vs;
39 
40  // Get EW charges.
41  const vector<double> Bq = fEWCharges(Q);
42 
43  // Tabulate input TMDs in the impact parameter to make the
44  // integral faster.
45  const auto TabFunc = [] (double const& b) -> double{ return log(b); };
46  const auto InvTabFunc = [] (double const& fb) -> double{ return exp(fb); };
47  const TabulateObject<Set<Distribution>> TabulatedTMDs{[&] (double const& b) -> Set<Distribution>
48  { return EvolvedTMDPDFs(b, muf, zetaf); }, 50, 5e-5, 30, 3, {}, TabFunc, InvTabFunc};
49 
50  // Construct the TMD luminosisty in b scale to be fed to be
51  // trasformed in qT space.
52  const auto TMDLumib = [=] (double const& b) -> double
53  {
54  // Get map of the TMDs in "x1" and "x2" and rotate them into the
55  // physical basis.
56  map<int,double> TMD1 = QCDEvToPhys(TabulatedTMDs.EvaluateMapxQ(x1, b));
57  map<int,double> TMD2 = QCDEvToPhys(TabulatedTMDs.EvaluateMapxQ(x2, b));
58 
59  // Construct the combination of TMDs weighted by the EW
60  // charges.
61  double lumi = 0;
62  for (int i = 1; i <= 6; i++)
63  lumi += Bq[i-1] * ( TMD1[i] * TMD2[-i] + TMD1[-i] * TMD2[i] );
64 
65  // Divide by x1 and x2 because the "EvolvedTMDPDFs" function
66  // returns x times the TMDs.
67  lumi /= x1 * x2;
68 
69  return lumi;
70  };
71 
72  // Compute hard coefficient, including the other kinematic
73  // factors.
74  const double hcs = ConvFact * FourPi * HardCrossSectionDY(PerturbativeOrder, Alphas(muf), NF(muf, Thresholds), muf/Q) / 9 / pow(Vs*Q,2);
75 
76  return [TMDLumib,hcs,Q] (double const& qT) -> double
77  {
78  const double cutPref = 1 + pow(qT/Q,2) / 2;
79  return hcs * cutPref * OgataQuadrature(TMDLumib, qT);
80  // For test purposes, one can comment out the line above, in
81  // which the Ogata quadrature is used, and comment in the two
82  // lines below to use the DGauss quadrature. The second will
83  // be substantially slower but should return consistent
84  // results.
85  //const Integrator integrand{[=] (double const& bT) -> double{ return TMDLumib(bT) * j0(qT * bT) * bT / 2; }};
86  //return hcs * cutPref * integrand.integrate(0.00005,30,eps9);
87  };
88  }

91  function<double(double const&)> TmdCrossSectionSIDIS(double const& Vs,
92  double const& x,
93  double const& y,
94  double const& z,
95  function<Set<Distribution>(double const&, double const&, double const&)> const& EvolvedTMDPDFs,
96  function<Set<Distribution>(double const&, double const&, double const&)> const& EvolvedTMDFFs,

104  // Compute virtuality of the photon.
105  const double Q = sqrt( x * y ) * Vs;
106

108  // to Q and Q^2, respectively, through "cmuf" and "czetaf".

116  // the integral faster.

118  const auto InvTabFunc = [] (double const& fb) -> double{ return exp(fb); };
119  const TabulateObject<Set<Distribution>> TabulatedPDFs{[&] (double const& b) -> Set<Distribution>
120  { return EvolvedTMDPDFs(b, muf, zetaf); }, 50, 5e-5, 30, 3, {}, TabFunc, InvTabFunc};
121  const TabulateObject<Set<Distribution>> TabulatedFFs{[&] (double const& b) -> Set<Distribution>
122  { return EvolvedTMDFFs(b, muf, zetaf); }, 50, 1e-2, 30, 3, {}, TabFunc, InvTabFunc};
123

125  // trasformed in qT space.
126  const auto TMDLumib = [=] (double const& b) -> double
127  {
128  // Get map of the TMDs in "x1" and "x2" and rotate them into the
129  // physical basis.
130  map<int,double> PDFs = QCDEvToPhys(TabulatedPDFs.EvaluateMapxQ(x, b));
131  map<int,double> FFs = QCDEvToPhys(TabulatedFFs.EvaluateMapxQ(z, b));
132 
133  // Construct the combination of TMDs weighted by the EW
134  // charges.
135  double lumi = 0;
136  for (int i = 1; i <= 6; i++)
137  lumi += Bq[i-1] * ( PDFs[i] * FFs[i] + PDFs[-i] * FFs[-i] );
138 
139  // Divide by x1 and x2 because the "EvolvedTMDPDFs" function
140  // returns x times the TMDs.
141  lumi /= x * z;
142 
143  return lumi;
144  };
145

147  // factors.
148  const double hcs = 2 * M_PI * HardCrossSectionSIDIS(PerturbativeOrder, Alphas(muf), NF(muf, Thresholds), muf/Q) / x / y / pow(Q,2);
149  //const double hcs = ConvFact * 2 * M_PI * HardCrossSectionSIDIS(PerturbativeOrder, Alphas(muf), NF(muf, Thresholds), muf/Q) / x / y / pow(Q,2);
150

152  {
153  return hcs * OgataQuadrature(TMDLumib, qT);
154  // For test purposes, one can comment out the line above, in
155  // which the Ogata quadrature is used, and comment in the two
156  // lines below to use the DGauss quadrature. The second will
157  // be substantially slower but should return consistent
158  // results.
159  //const Integrator integrand{[=] (double const& bT) -> double{ return TMDLumib(bT) * j0(qT * bT) * bT / 2; }};

161  };
162  }

165  function<double(double const&)> TmdCrossSectionDY(double const& Vs,
166  double const& Qmin,
167  double const& Qmax,
168  double const& ymin,
169  double const& ymax,
170  function<Set<Distribution>(double const&)> const& InTMDPDFs,
171  function<vector<double>(double const&, double const&, double const&)> const& EvolFact,
172  function<double(double const&)> const& Alphas,
173  function<vector<double>(double const&)> const& fEWCharges,
174  int const& PerturbativeOrder,
175  vector<double> const& Thresholds,
176  double const& cmuf,
177  double const& czetaf,
178  double const& IntEps)
179  {
180  // Tabulation function and its inverse.
181  const auto TabFunc = [] (double const& b) -> double{ return log(b); };
182  const auto InvTabFunc = [] (double const& fb) -> double{ return exp(fb); };
183 
184  // Tabulate initial TMD PDFs.
185  const TabulateObject<Set<Distribution>> TabInTMDs{[&] (double const& b) -> Set<Distribution>
186  { return InTMDPDFs(b); }, 50, 5e-5, 30, 3, {}, TabFunc, InvTabFunc};
187 
188  // If integration over Q and y is required, it is advantageous to
189  // perform these integrations before the integral over the impact
190  // parameter b (Fourier transform).
191  return [=] (double const& qT) -> double
192  {
193  const Integrator Qintegrand{
194  [&] (double const& lnQ) -> double
195  {
196  const double Q = exp(lnQ);
197  // Compute "muf" and "zetaf". They are assumed to be
198  // proportional to Q and Q^2, respectively, through
199  // "cmuf" and "czetaf".
200  const double muf = cmuf * Q;
201  const double zetaf = czetaf * Q * Q;
202 
203  // Get EW charges.
204  const vector<double> Bq = fEWCharges(Q);
205 
206  // Tabulate input TMDs in the impact parameter to make the
207  // integral faster.
208  const TabulateObject<Set<Distribution>> TabEvTMDs{[&] (double const& b) -> Set<Distribution>
209  { return EvolFact(b, muf, zetaf) * TabInTMDs.Evaluate(b); }, 50, 5e-5, 30, 3, {}, TabFunc, InvTabFunc};
210 
211  const auto bintegrand = [&] (double const& b) -> double
212  {
213  // Define y integrand.
214  const Integrator yintegrand{
215  [&] (double const& ey) -> double
216  {
217  // Compute values of x1 and x2.
218  const double x1 = Q / ey / Vs;
219  const double x2 = Q * ey / Vs;
220 
221  // Get map of the TMDs in "x1" and "x2" and
222  // rotate them into the physical basis.
223  map<int,double> TMD1 = QCDEvToPhys(TabEvTMDs.EvaluateMapxQ(x1, b));
224  map<int,double> TMD2 = QCDEvToPhys(TabEvTMDs.EvaluateMapxQ(x2, b));
225 
226  // Construct the combination of TMDs weighted by
227  // the EW charges.
228  double lumi = 0;
229  for (int i = 1; i <= 6; i++)
230  lumi += Bq[i-1] * ( TMD1[i] * TMD2[-i] + TMD1[-i] * TMD2[i] );
231 
232  // Divide by x1 and x2 because "EvolvedTMDPDFs"
233  // returns x times the TMD PDFs.
234  lumi /= x1 * x2 * ey ;
235 
236  return lumi;
237  }
238  };
239 
240  // Integrate in y over [ymin:ymax] or return the
241  // value if ymin = ymax.
242  if (ymin == ymax)
243  return yintegrand.integrand(exp(ymin)) * exp(ymin);
244  else
245  return yintegrand.integrate(exp(ymin), exp(ymax), IntEps);
246  };
247 
248  const double cutPref = 1 + pow(qT/Q,2) / 2;
249  const double HardFact = HardCrossSectionDY(PerturbativeOrder, Alphas(muf), NF(muf, Thresholds), cmuf);
250  return 2 * cutPref * HardFact * OgataQuadrature(bintegrand, qT);
251  }
252  };
253  // Integrate in Q over [Qmin:Qmax] or return the
254  // value if Qmin = Qmax.
255  if (Qmin == Qmax)
256  return ConvFact * FourPi * Qintegrand.integrand(log(Qmin)) / 9 / Vs / Vs / Qmin;
257  else
258  return ConvFact * FourPi * Qintegrand.integrate(log(Qmin), log(Qmax), IntEps) / 9 / Vs / Vs;
259  };
260  }

263  double HardCrossSectionDY(int const& pt, double const& Alphas, int const& nf, double const& kappa)
264  {
265  // Compute log anf its powers.
266  const double lQ2 = 2 * log(kappa);
267  const double lQ22 = lQ2 * lQ2;
268  const double lQ23 = lQ22 * lQ2;
269  const double lQ24 = lQ23 * lQ2;
270 
271  // Compute coupling and its powers.
272  const double as = Alphas / FourPi;
273  const double as2 = as * as;
274 
275  // Now compute hard cross section according to the perturbative
276  // order.
277  double hxs = 1;
278  if (pt > 0)
279  hxs += 2 * as * CF * ( - lQ22 - 3 * lQ2 - 8 + 7 * Pi2 / 6 );
280  if (pt > 1)
281  hxs += 2 * as2 * CF *
282  ( CF * ( lQ24 + 6 * lQ23 + ( 25. - 7 * Pi2 / 3 ) * lQ22 + ( 93. / 2. - 5 * Pi2 - 24 * zeta3 ) * lQ2
283  + 511. / 8. - 83 * Pi2 / 6 - 30 * zeta3 + 67 * Pi2 * Pi2 / 60 ) +
284  CA * ( - 11 * lQ23 / 9 + ( - 233. / 18. + Pi2 / 3 ) * lQ22 + ( - 2545. / 54. + 22 * Pi2 / 9 + 26 * zeta3 ) * lQ2
285  - 51157. / 648. + 1061 * Pi2 / 108 + 313 * zeta3 / 9 - 4 * Pi2 * Pi2 / 45 ) +
286  TR * nf * ( 4 * lQ23 / 9 + 38 * lQ22 / 9 + ( 418. / 27. - 8 * Pi2 / 9 ) * lQ2
287  + 4085. / 162. - 91 * Pi2 / 27 + 4 * zeta3 / 9 ) );
288 
289  return hxs;
290  };
