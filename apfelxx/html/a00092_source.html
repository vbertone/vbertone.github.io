21  SubGrid::SubGrid(int const& nx, double const& xMin, int const& InterDegree):
22  _nx(nx),
23  _InterDegree(InterDegree),
24  _IsExternal(false),
25  _xMin(xMin),
26  _xMax(1)
27  {
28  // Compute grid.
29  _Step = log(_xMax / _xMin) / _nx;
30 
31  // Building log spaced grid in x. Number of points in x + 1 (bins)
32  // + extra nodes for rhs interpolation.
33  _xsg.resize(_nx+_InterDegree+1, 0);
34 
35  _xsg[0] = _xMin;
36  double const exps = exp(_Step);
37  for (int ix = 1; ix < (int) _xsg.size(); ix++)
38  _xsg[ix] = _xsg[ix-1] * exps;
39  _xsg[_nx] = 1;
40 
41  _lxsg.resize(_xsg.size());
42  for (int ix = 0; ix < (int) _xsg.size(); ix++)
43  _lxsg[ix] = log(_xsg[ix]);
44  }

47  SubGrid::SubGrid(vector<double> const& xsg, int const& InterDegree):
48  _nx(xsg.size()-1),
49  _InterDegree(InterDegree),
50  _IsExternal(true),
51  _xMin(xsg[0]),
52  _xMax(1),
53  _Step(0)
54  {
55  _xsg.resize(_nx+InterDegree+1, 0);
56  copy(xsg.begin(), xsg.end(), _xsg.begin());
57 
58  // Check that the last point of the user-given grid is equal to
59  // one.
60  if (abs(_xsg[_nx]-1) >= eps11)
61  throw runtime_error(error("SubGrid::SubGrid","The upper value of the external grid does not coincide with 1."));
62  else
63  _xsg[_nx] = 1;
64 
65  // Extend the grid for x > 1 for interpolation reasons using the
66  // same width of the last bin in log scale.
67  double const step = - log( xsg[_nx-1] );
68  double const exps = exp(step);
69  for (int ix = _nx; ix < (int) _xsg.size(); ix++)
70  _xsg[ix] = _xsg[ix-1] * exps;
71 
72  _lxsg.resize(_xsg.size());
73  for (int ix = 0; ix < (int) _xsg.size(); ix++)
74  _lxsg[ix] = log(_xsg[ix]);
75  }

78  bool SubGrid::operator == (SubGrid const& sg) const
79  {
80  // Are both external or internal grids?
81  if (_IsExternal != sg._IsExternal)
82  return false;
83 
84  // In case they are external ...
85  if (_IsExternal)

87  if (_xsg != sg._xsg)
88  return false;
89  if (_InterDegree != sg._InterDegree)
90  return false;
91  }
92  // In case they are internal ...
93  else
94  {
95  if (_nx != sg._nx)
96  return false;
97  if (_xMin != sg._xMin)
98  return false;
99  if (_xMax != sg._xMax)
100  return false;
101  if (_InterDegree != sg._InterDegree)
102  return false;
103  }
104  return true;
105  }
