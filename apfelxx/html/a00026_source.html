104  const double ln1mx3 = ln1mx2 * ln1mx;

106  const double ln1px2 = ln1px * ln1px;

109  const double a01 =
110  ( 1 - 2 * x + 2 * x2 ) * ( 8 * zeta3 + 4 * ln1mx3 / 3 - 8 * ln1mx * S111mx
111  + 8 * zeta2 * lnx - 4 * lnx * ln1mx2 + 2 * lnx3 / 3
112  - 8 * lnx * S111mx + 8 * S211mx - 24 * S121mx );
113  const double b01 =
114  - ( 4 + 96 * x - 64 * x2 ) * S111mx
115  - ( 4 - 48 * x + 40 * x2 ) * zeta2
116  - ( 8 + 48 * x - 24 * x2 ) * lnx * ln1mx

120  - ( 16 + 18 * x + 48 * x2 ) * lnx
121  + 26 - 82 * x + 80 * x2
122  + x2 * ( - 16 * zeta2 * lnx + 4 * lnx3 / 3 + 16 * lnx * S111mx + 32 * S121mx );

125  ( 1 - 2 * x + 2 * x2 ) * ( - 4 * ln1mx3 / 3 + 8 * ln1mx * S111mx - 8 * S211mx )
126  + ( 1 + 2 * x + 2 * x2 ) * ( - 8 * zeta2 * ln1px - 16 * ln1px * S11mx - 8 * lnx * ln1px2
127  + 4 * lnx2 * ln1px + 8 * lnx * S11mx - 8 * S21mx - 16 * S12mx )

129  - ( 4 + 8 * x ) * lnx3 / 3
130  + ( 8 - 32 * x + 16 * x2 ) * zeta3
131  - ( 16 + 64 * x ) * zeta2 * lnx;
132  const double b02 =
133  ( 16 * x + 16 * x2 ) * ( S11mx + lnx * ln1px ) // There is a typo here in the e-Print ( 16 -> 16x )
134  + ( 32 / x / 3 + 12 + 64 * x - 272 * x2 / 3 ) * S111mx
135  - ( 12 + 48 * x - 260 * x2 / 3 + 32 / x / 3 ) * zeta2
