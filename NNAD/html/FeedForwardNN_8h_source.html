56  FeedForwardNN(std::vector<int> const &Arch,
57  int const &RandomSeed,
58  bool const &Report = false,
59  std::function<T(T const &)> const &ActFun = Sigmoid<T>,
60  std::function<T(T const &)> const &dActFun = dSigmoid<T>,
61  bool const &LinearOutput = true):
62  _Arch(Arch),
63  _ActFun(ActFun),
64  _dActFun(dActFun),
65  _LinearOutput(LinearOutput)
66  {
67  // Number of layers
68  const int nl = (int)_Arch.size();
69 
70  // Store sub seeds
71  srand(RandomSeed);
72  std::vector<int> sub_seeds;
73  for (int l = 1; l < nl; l++)
74  {
75  sub_seeds.push_back(rand());
76  sub_seeds.push_back(rand());
77  }
78 
79  // Number of parameters
80  _Np = 0;
81 
82  // Initialise links and biases with random numbers. Biases are
83  // treated as matrices that connect consecutive layers with as many
84  // rows as nodes of the rightmost layer and as many columns are the
85  // leftmost layer. Biases are treated as one-column matrices
86  // associated to each layer (but the input one) with as many rows as
87  // nodes.
88  int iseed = 0;
89  for (int l = 1; l < nl; l++)
90  {
91  _Links.insert({l, Matrix<T>{_Arch[l], _Arch[l - 1], sub_seeds.at(iseed++)}});
92  _Biases.insert({l, Matrix<T>{_Arch[l], 1, sub_seeds.at(iseed++)}});
93  _Np += _Arch[l] * (_Arch[l - 1] + 1);
94  }
95 
96  // Construct maps that index the free parameters. Each parameter is
97  // associated to an integer and a string.
98  int count = 0;
99  for (int l = nl - 1; l > 0; l--)
100  for (int i = 0; i < _Arch[l]; i++)
101  {
102  // Biases
103  const coordinates coord = std::make_tuple(false, l, i, 0);
104  const std::string name = "B_" + std::to_string(l) + "_" + std::to_string(i);
105  _StrIntMap.insert({name, count});
106  _IntMap.insert({count++, coord});
107  _StrMap.insert({name, coord});
108 
109  // Links
110  for (int j = 0; j < _Arch[l - 1]; j++)
111  {
112  const coordinates coord = std::make_tuple(true, l, i, j);
113  const std::string name = "L_" + std::to_string(l) + "_" + std::to_string(i) + "_" + std::to_string(j);
114  _StrIntMap.insert({name, count});
115  _IntMap.insert({count++, coord});
116  _StrMap.insert({name, coord});
117  }
118  }
119 
120  // Report NN parameters
121  if (Report)
122  {
123  std::cout << "\nFeed-forward neural network initialised:" << std::endl;
124  std::cout << "- architecture = [ ";
125  for (auto const &n : _Arch)
126  std::cout << n << " ";
127  std::cout << "]" << std::endl;
128  std::cout << "- number of parameters = " << _Np << std::endl;
129  if (_LinearOutput)
130  std::cout << "- linear output activation function" << std::endl;
131  std::cout << "\n";
132  }
133  }

269  std::vector<T> GetParameters() const
270  {
271  // Initialise output vector.
272  std::vector<T> output(_Np);
273 
274  // Element counter
275  int count = 0;
276 
277  // Number of layers
278  const int nl = (int)_Arch.size();
279 
280  // Now run backwards on the layers to compute the derivatives. Same
281  // ordering as in the 'Derive' function.
282  for (int l = nl - 1; l > 0; l--)
283  for (int i = 0; i < _Arch[l]; i++)
284  {
285  // Biases
286  output[count++] = _Biases.at(l).GetElement(i, 0);
287 
288  // Links
289  for (int j = 0; j < _Arch[l - 1]; j++)
290  output[count++] = _Links.at(l).GetElement(i, j);
291  }
292 
293  return output;

297  std::vector<T> Evaluate(std::vector<T> const &Input) const
298  {
299  // Check that the size of the input vector is equal to the number of
300  // input nodes.
301  if ((int)Input.size() != _Arch[0])
302  Error("Evaluate: the number of inputs does not match the number of input nodes.");
303 
304  // Number of layers
305  const int nl = (int)_Arch.size();
306 
307  // Construct output of the NN recursively for the hidden layers.
308  Matrix<T> y(Input.size(), 1, Input);
309  for (int l = 1; l < nl - 1; l++)
310  y = Matrix<T>(_Links.at(l) * y + _Biases.at(l), _ActFun);
311 
312  // Now take care of the output layer according to whether it is
313  // linear or not.
314  y = Matrix<T>(_Links.at(nl - 1) * y + _Biases.at(nl - 1), (_LinearOutput ? [](T const &x) -> T { return x; } : _ActFun));
315 
316  return y.GetVector();
317  }

320  std::vector<T> Derive(std::vector<T> const &Input) const
321  {
322  // Check that the size of the input vector is equal to the number of
323  // input nodes.
324  if ((int)Input.size() != _Arch[0])
325  Error("Derive: the number of inputs does not match the number of input nodes.");
326 
327  // Initialise output vector.
328  std::vector<T> output((_Np + 1) * _Arch.back());
329 
330  // Element counter
331  int count = 0;
332 
333  // Number of layers
334  const int nl = (int)_Arch.size();
335 
336  // Compute NN recursively and save the quantities that will be
337  // needed to compute the derivatives.
338  std::map<int, Matrix<T>> y;
339  std::map<int, Matrix<T>> z;
340  y.insert({0, Matrix<T>{(int) Input.size(), 1, Input}});
341  for (int l = 1; l < nl - 1; l++)
342  {
343  const Matrix<T> M = _Links.at(l) * y.at(l - 1) + _Biases.at(l);
344  y.insert({l, Matrix<T>{M, _ActFun}});
345  z.insert({l, Matrix<T>{M, _dActFun}});
346  }
347 
348  // Now take care of the output layer according to whether it is
349  // linear or not.
350  const Matrix<T> M = _Links.at(nl - 1) * y.at(nl - 2) + _Biases.at(nl - 1);
351  y.insert({nl - 1, Matrix<T>{M, (_LinearOutput ? [](T const &x) -> T { return T(x); } : _ActFun)}});
352  z.insert({nl - 1, Matrix<T>{M, (_LinearOutput ? [](T const &) -> T { return T(1); } : _dActFun)}});
353 
354  // Output vector of vector. Push back NN itself as a first element.
355  for (auto const &e : y.at(nl - 1).GetVector())
356  output[count++] = e;
357 
358  // First compute the Sigma matrix on the output layer that is just
359  // the unity matrix.
360  Matrix<T> Sigma(_Arch[nl - 1], _Arch[nl - 1], std::vector<T>(_Arch[nl - 1] * _Arch[nl - 1], T(0.)));
361  for (int k = 0; k < _Arch[nl - 1]; k++)
362  Sigma.SetElement(k, k, T(1));
363 
364  // Now run backwards on the layers to compute the derivatives.
365  for (int l = nl - 1; l > 0; l--)
366  {
367  const std::vector<T> zl = z.at(l).GetVector();
368  const std::vector<T> ylmo = y.at(l - 1).GetVector();
369  for (int i = 0; i < _Arch[l]; i++)
370  {
371  // Compute derivatives w.r.t. the biases
372  for (int k = 0; k < _Arch[nl - 1]; k++)
373  output[count++] = Sigma.GetElement(k, i) * zl[i];
374 
375  // Compute derivatives w.r.t. the links
376  for (int j = 0; j < _Arch[l - 1]; j++)
377  for (int k = 0; k < _Arch[nl - 1]; k++)
378  output[count++] = Sigma.GetElement(k, i) * zl[i] * ylmo[j];
379  }
380 
381  // Compute Matrix "S" on this layer.
382  std::vector<T> entries;
383  for (int i = 0; i < _Arch[l]; i++)
384  for (int j = 0; j < _Arch[l - 1]; j++)
385  entries.push_back(zl[i] * _Links.at(l).GetElement(i, j));
386 
387  const Matrix<T> S{_Arch[l], _Arch[l - 1], entries};
388 
389  // Update Matrix "Sigma" for the next step.
390  Sigma = Sigma * S;
391  }
392 
393  return output;
394  }
